```typescript
2. Calculate threat probabilities using historical data and threat intelligence
3. Analyze potential impacts on business and technical operations
4. Calculate risk scores using the standard risk formula
5. Generate comprehensive risk reports with prioritized recommendations

Always base assessments on factual data and industry best practices.`,
    };

    try {
      // 1. 获取待评估资产
      const assets = await this.getAssets(context);
      this.reportProgress(10, `Found ${assets.length} assets to assess`);

      // 2. 获取相关威胁
      const threats = await this.getThreats(context);
      this.reportProgress(20, `Found ${threats.length} relevant threats`);

      // 3. 评估每个资产
      for (let i = 0; i < assets.length; i++) {
        const asset = assets[i];
        this.reportProgress(
          20 + (i / assets.length) * 60,
          `Assessing asset: ${asset.name}`
        );

        // 3.1 资产估值
        const assetValuation = await this.assetValuator.valuate(asset);

        // 3.2 对每个威胁计算风险
        for (const threat of threats) {
          // 威胁概率
          const probability = await this.threatProbability.calculate(threat, asset);

          // 影响分析
          const impact = await this.impactAnalyzer.analyze(asset, threat);

          // 风险计算
          const riskScore = await this.riskCalculator.calculate({
            asset,
            assetValuation,
            threat,
            probability,
            impact,
          });

          assessments.push({
            asset,
            threat,
            assetValuation,
            probability,
            impact,
            riskScore,
            timestamp: new Date(),
          });
        }
      }

      // 4. 风险优先级排序
      this.reportProgress(85, 'Prioritizing risks...');
      const prioritizedRisks = this.riskCalculator.prioritizeRisks(assessments);

      // 5. 生成报告
      this.reportProgress(95, 'Generating risk report...');
      const report = await this.reportGenerator.generate(prioritizedRisks);

      // 6. 更新指标
      this.updateMetrics(assets.length, assessments.length, prioritizedRisks);

      return {
        success: true,
        data: {
          report,
          assessmentsCount: assessments.length,
          highRisksCount: prioritizedRisks.high.length,
          mediumRisksCount: prioritizedRisks.medium.length,
          lowRisksCount: prioritizedRisks.low.length,
        },
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    } catch (error) {
      await this.handleError(error as Error);
      return {
        success: false,
        error: {
          code: 'RISK_ASSESSMENT_FAILED',
          message: (error as Error).message,
          recoverable: true,
        },
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    }
  }
```

### risk-calculator.ts 骨架

```typescript
import type { RiskAssessment, RiskScore, RiskMatrix } from './types';

export class RiskCalculator {
  /**
   * 计算风险评分
   * 风险 = 资产价值 × 威胁概率 × 脆弱性 × 影响系数
   */
  async calculate(assessment: RiskAssessment): Promise<RiskScore> {
    const assetValue = assessment.assetValuation.totalScore; // 1-10
    const threatProbability = assessment.probability; // 0-1
    const vulnerability = assessment.impact.vulnerabilityScore || 0.5; // 0-1
    const impactFactor = assessment.impact.impactFactor; // 0-1

    // 计算风险评分 (0-100)
    const rawScore = assetValue * threatProbability * vulnerability * impactFactor * 10;
    const score = Math.min(100, Math.max(0, rawScore));

    // 确定风险等级
    let level: 'critical' | 'high' | 'medium' | 'low';
    if (score >= 80) level = 'critical';
    else if (score >= 60) level = 'high';
    else if (score >= 40) level = 'medium';
    else level = 'low';

    return {
      score,
      level,
      components: {
        assetValue,
        threatProbability,
        vulnerability,
        impactFactor,
      },
    };
  }

  /**
   * 风险优先级排序
   */
  prioritizeRisks(assessments: RiskAssessment[]): {
    critical: RiskAssessment[];
    high: RiskAssessment[];
    medium: RiskAssessment[];
    low: RiskAssessment[];
    all: RiskAssessment[];
  } {
    const sorted = assessments.sort((a, b) => b.riskScore.score - a.riskScore.score);

    return {
      critical: sorted.filter(a => a.riskScore.level === 'critical'),
      high: sorted.filter(a => a.riskScore.level === 'high'),
      medium: sorted.filter(a => a.riskScore.level === 'medium'),
      low: sorted.filter(a => a.riskScore.level === 'low'),
      all: sorted,
    };
  }

  /**
   * 生成风险矩阵
   */
  generateRiskMatrix(assessments: RiskAssessment[]): RiskMatrix {
    const matrix: RiskMatrix = {
      dimensions: {
        x: 'probability',
        y: 'impact',
      },
      cells: [],
    };

    // 5x5 风险矩阵
    for (let prob = 1; prob <= 5; prob++) {
      for (let impact = 1; impact <= 5; impact++) {
        const cellAssessments = assessments.filter(a => {
          const probLevel = Math.ceil(a.probability * 5);
          const impactLevel = Math.ceil(a.impact.impactFactor * 5);
          return probLevel === prob && impactLevel === impact;
        });

        matrix.cells.push({
          probability: prob,
          impact: impact,
          riskLevel: this.getCellRiskLevel(prob, impact),
          assessments: cellAssessments,
        });
      }
    }

    return matrix;
  }

  private getCellRiskLevel(prob: number, impact: number): 'critical' | 'high' | 'medium' | 'low' {
    const score = prob * impact;
    if (score >= 20) return 'critical';
    if (score >= 12) return 'high';
    if (score >= 6) return 'medium';
    return 'low';
  }
}
```

## 风险计算公式

```
风险 = 资产价值 × 威胁概率 × 脆弱性 × 影响系数

其中：
- 资产价值: 1-10 (CIA三元组评分)
- 威胁概率: 0-1 (基于历史数据和威胁情报)
- 脆弱性: 0-1 (基于漏洞扫描结果)
- 影响系数: 0-1 (业务影响评估)

风险评分等级：
- Critical: 80-100
- High: 60-79
- Medium: 40-59
- Low: 0-39
```
