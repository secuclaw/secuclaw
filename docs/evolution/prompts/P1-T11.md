### server.ts 骨架

```typescript
import http, { type IncomingMessage, type ServerResponse } from "node:http";
import type { Socket } from "node:net";
import chokidar from "chokidar";
import { WebSocketServer, type WebSocket } from "ws";
import { resolveStateDir } from "../config/paths.js";
import { injectLiveReload } from "./live-reload.js";
import { resolveFileWithinRoot } from "./file-resolver.js";

export type CanvasServerOpts = {
  runtime: RuntimeEnv;
  rootDir?: string;
  port?: number;
  liveReload?: boolean;
};

export async function startCanvasServer(opts: CanvasServerOpts): Promise<CanvasServer> {
  const rootDir = opts.rootDir ?? resolveDefaultCanvasRoot();
  const port = opts.port ?? 0;
  
  // WebSocket for live reload
  const wss = opts.liveReload ? new WebSocketServer({ noServer: true }) : null;
  const sockets = new Set<WebSocket>();
  
  if (wss) {
    wss.on("connection", (ws) => {
      sockets.add(ws);
      ws.on("close", () => sockets.delete(ws));
    });
  }
  
  // File watcher for hot reload
  const watcher = opts.liveReload
    ? chokidar.watch(rootDir, { ignoreInitial: true })
    : null;
  
  watcher?.on("all", () => {
    for (const ws of sockets) {
      try { ws.send("reload"); } catch {}
    }
  });
  
  // HTTP server
  const server = http.createServer(async (req, res) => {
    await handleCanvasRequest(req, res, rootDir, opts.liveReload);
  });
  
  // WebSocket upgrade
  server.on("upgrade", (req, socket, head) => {
    if (wss && req.url === "/__secuclaw__/ws") {
      wss.handleUpgrade(req, socket as Socket, head, (ws) => {
        wss.emit("connection", ws, req);
      });
    }
  });
  
  await new Promise<void>((resolve) => server.listen(port, resolve));
  
  return {
    port: (server.address() as any).port,
    rootDir,
    close: async () => {
      await watcher?.close();
      wss?.close();
      server.close();
    },
  };
}
```

### state.ts 状态管理

```typescript
export interface CanvasState {
  version: string;
  elements: Map<string, CanvasElement>;
  viewport: ViewportState;
  theme: 'light' | 'dark';
  lastUpdate: number;
}

export type StateListener = (state: CanvasState) => void;

export class CanvasStateManager {
  private state: CanvasState;
  private listeners = new Set<StateListener>();
  
  constructor(initialState?: Partial<CanvasState>) {
    this.state = {
      version: '1.0',
      elements: new Map(),
      viewport: { x: 0, y: 0, width: 1920, height: 1080, zoom: 1 },
      theme: 'dark',
      lastUpdate: Date.now(),
      ...initialState,
    };
  }
  
  getState(): CanvasState {
    return { ...this.state };
  }
  
  setState(updates: Partial<CanvasState>): void {
    this.state = { ...this.state, ...updates, lastUpdate: Date.now() };
    this.notify();
  }
  
  subscribe(listener: StateListener): () => void {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }
  
  private notify(): void {
    const state = this.getState();
    for (const listener of this.listeners) {
      listener(state);
    }
  }
}
```

## 注意事项

1. **性能优化**: 大量元素时需要虚拟化渲染，只渲染可视区域
2. **内存管理**: 及时清理不可见元素，避免内存泄漏
3. **WebSocket重连**: 客户端断开后需要自动重连
4. **文件安全**: file-resolver需要防止路径遍历攻击
5. **跨平台**: 需要支持iOS WebView和Android WebView的JS Bridge
