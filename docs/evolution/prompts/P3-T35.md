## 示例代码片段

### index.ts 骨架

```typescript
import { FileReader } from './reader';
import { FileWriter } from './writer';
import { FileScanner } from './scanner';
import { FileGuard } from './guard';

export class EdgeFileTool {
  private reader: FileReader;
  private writer: FileWriter;
  private scanner: FileScanner;
  private guard: FileGuard;

  constructor() {
    this.reader = new FileReader();
    this.writer = new FileWriter();
    this.scanner = new FileScanner();
    this.guard = new FileGuard();
  }

  /**
   * 读取文件
   */
  async read(path: string): Promise<string> {
    // 验证路径
    if (!this.isAllowed(path)) {
      throw new Error(`Access denied: ${path}`);
    }

    return await this.reader.read(path);
  }

  /**
   * 写入文件
   */
  async write(path: string, content: string): Promise<void> {
    // 验证路径
    if (!this.isAllowed(path)) {
      throw new Error(`Access denied: ${path}`);
    }

    await this.writer.write(path, content);
  }

  /**
   * 列出文件
   */
  async list(dir: string): Promise<string[]> {
    // 验证路径
    if (!this.isAllowed(dir)) {
      throw new Error(`Access denied: ${dir}`);
    }

    return await this.scanner.scan(dir);
  }

  /**
   * 检查权限
   */
  isAllowed(path: string): boolean {
    return this.guard.validate(path);
  }
}
```
