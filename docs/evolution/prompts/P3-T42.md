## 示例代码片段

### watcher.ts 骨架

```typescript
import { watch, FSWatcher } from 'fs';
import { promisify } from 'util';

export class ConfigWatcher {
  private watcher?: FSWatcher;
  private callbacks: ConfigChangeCallback[] = [];

  /**
   * 开始监视
   */
  watch(path: string): void {
    this.watcher = watch(path, (eventType, filename) => {
      if (eventType === 'change') {
        this.notifyChange(filename);
      }
    });
  }

  /**
   * 停止监视
   */
  unwatch(): void {
    if (this.watcher) {
      this.watcher.close();
      this.watcher = undefined;
    }
  }

  /**
   * 注册回调
   */
  onChange(callback: ConfigChangeCallback): void {
    this.callbacks.push(callback);
  }

  private notifyChange(filename: string): void {
    for (const callback of this.callbacks) {
      callback(filename);
    }
  }
}
```

import { ConfigValidator } from './validator';

export class HotReloader {
  private watcher: ConfigWatcher;
  private validator: ConfigValidator;
  private config: any;

  constructor() {
    this.watcher = new ConfigWatcher();
    this.validator = new ConfigValidator();
  }

  /**
   * 重新加载
   */
  async reload(path: string): Promise<void> {
    // 读取新配置
    const newConfig = await this.loadConfig(path);
    
    // 验证配置
    const validation = this.validator.validate(newConfig);
    if (!validation.valid) {
      throw new Error(`Invalid config: ${validation.errors.join(', ')}`);
    }
    
    // 应用配置
    await this.apply(newConfig);

## 热更新流程

```
┌─────────────┐     ┌─────────────┐     ┌─────────────┐     ┌─────────────┐
│ 文件变更    │ ──► │ 读取配置    │ ──► │ 验证配置    │ ──► │ 应用配置    │
└─────────────┘     └─────────────┘     └─────────────┘     └─────────────┘
                                              │                    │
                                              ▼                    ▼
                                        ┌─────────────┐     ┌─────────────┐
                                        │ 验证失败    │     │ 通知模块    │
                                        │ 回滚配置    │     │ 重新加载    │
                                        └─────────────┘     └─────────────┘
```

---

## 注意事项

1. **文件监视**: 及时检测配置变更
2. **验证优先**: 验证通过后再应用
3. **原子更新**: 保证配置一致性
4. **回滚机制**: 失败时自动回滚
