## 示例代码片段

### client.ts 骨架

```typescript
import { NgrokAPI } from './api';

export class NgrokClient {
  private api: NgrokAPI;
  private url?: string;

  constructor(apiKey: string) {
    this.api = new NgrokAPI(apiKey);
  }

  /**
   * 连接
   */
  async connect(port: number): Promise<string> {
    const tunnel = await this.api.createTunnel({
      addr: port,
      proto: 'http',
    });
    
    this.url = tunnel.public_url;
    return this.url;
  }

  /**
   * 断开
   */
  async disconnect(): Promise<void> {
    if (this.url) {
      await this.api.deleteTunnel(this.url);
      this.url = undefined;
    }
  }

  /**
   * 获取URL
   */
  getUrl(): string {
    if (!this.url) {
      throw new Error('Not connected');
    }
    return this.url;
  }
}
```
