## 示例代码片段

### talk.ts 骨架

```typescript
import { VoiceActivityDetector } from './vad';
import { ConversationManager } from './conversation-manager';
import type { SpeechCallback, ResponseCallback } from './types';

export class TalkMode {
  private vad: VoiceActivityDetector;
  private conversation: ConversationManager;
  private speechCallbacks: Set<SpeechCallback> = new Set();
  private responseCallbacks: Set<ResponseCallback> = new Set();
  private isRunning = false;
  private isContinuous = false;
  
  constructor() {
    this.vad = new VoiceActivityDetector();
    this.conversation = new ConversationManager();
  }

  async start(): Promise<void> {
    if (this.isRunning) return;
    this.isRunning = true;
    
    this.vad.onSpeechStart(() => this.handleSpeechStart());
    this.vad.onSpeechEnd((audio) => this.handleSpeechEnd(audio));
    
    await this.vad.start();
  }

  async stop(): Promise<void> {
    this.isRunning = false;
    await this.vad.stop();
  }

  interrupt(): void {
    // 中断当前AI回复
  }

  onUserSpeech(callback: SpeechCallback): void {
    this.speechCallbacks.add(callback);
  }

  setContinuousMode(enabled: boolean): void {
    this.isContinuous = enabled;
  }

  private handleSpeechStart(): void {
    // 语音开始处理
  }

  private async handleSpeechEnd(audio: Buffer): void {
    // 语音结束，转文字并处理
    for (const callback of this.speechCallbacks) {
      await callback(audio);
    }
  }
}
```
