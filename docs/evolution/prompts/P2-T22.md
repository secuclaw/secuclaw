### index.ts 骨架

```typescript
import { BaseHand, HandContext, HandResult, HandDefinition } from '../base';
import { PlaybookExecutor } from './playbook-executor';
import { AlertCorrelator } from './alert-correlator';
import { EvidenceCollector } from './evidence-collector';
import { ReportGenerator } from './report-generator';
import type { Incident, Playbook, Alert, Evidence, IncidentReport } from './types';

export class IncidentResponderHand extends BaseHand {
  private playbookExecutor: PlaybookExecutor;
  private alertCorrelator: AlertCorrelator;
  private evidenceCollector: EvidenceCollector;
  private reportGenerator: ReportGenerator;

  constructor() {
    const definition: HandDefinition = {
      id: 'incident-responder',
      name: 'Incident Responder Hand',
      description: 'Automated security incident response with playbook execution',
      category: 'security',
      version: '1.0.0',
      requirements: [],
      settings: [],
      metrics: [
        { label: 'Incidents Responded', memoryKey: 'incidents_count', format: 'number' },
        { label: 'Average Response Time', memoryKey: 'avg_response_time', format: 'duration' },
        { label: 'Playbooks Executed', memoryKey: 'playbooks_count', format: 'number' },
      ],
      tools: [
        'playbook_execute',
        'alert_correlate',
        'evidence_collect',
        'report_generate',
        'memory_store',
        'memory_recall',
      ],
      systemPrompt: `You are an automated incident response agent. Your role is to:
1. Analyze incoming security alerts and incidents
2. Execute appropriate response playbooks
3. Correlate related alerts to identify attack patterns
4. Collect forensic evidence for investigation
5. Generate comprehensive incident reports

Always follow the principle of "do no harm" - prioritize containment over remediation.`,
    };

    super(definition);
    this.playbookExecutor = new PlaybookExecutor();
    this.alertCorrelator = new AlertCorrelator();
    this.evidenceCollector = new EvidenceCollector();
    this.reportGenerator = new ReportGenerator();
  }

  async initialize(): Promise<void> {
    await this.playbookExecutor.initialize();
    await this.alertCorrelator.initialize();
    await this.evidenceCollector.initialize();
    await this.reportGenerator.initialize();
  }

  async execute(context: HandContext): Promise<HandResult> {
    const startTime = Date.now();

    try {
      // 1. 获取待处理事件
      const incidents = await this.getPendingIncidents(context);
      
      for (const incident of incidents) {
        this.reportProgress(10, `Processing incident: ${incident.id}`);
        
        // 2. 关联警报
        const correlatedAlerts = await this.alertCorrelator.correlate(incident.alerts);
        this.reportProgress(30, `Correlated ${correlatedAlerts.length} alerts`);
        
        // 3. 执行响应剧本
        const playbookResult = await this.playbookExecutor.execute(
          incident.playbookId,
          { incident, correlatedAlerts }
        );
        this.reportProgress(60, 'Playbook executed');
        
        // 4. 收集证据
        const evidence = await this.evidenceCollector.collect(incident);
        this.reportProgress(80, `Collected ${evidence.length} evidence items`);
        
        // 5. 生成报告
        const report = await this.reportGenerator.generate(incident, {
          correlatedAlerts,
          playbookResult,
          evidence,
        });
        
        // 6. 更新指标
        this.updateMetrics(incident, Date.now() - startTime);
      }

      return {
        success: true,
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    } catch (error) {
      await this.handleError(error as Error);
      return {
        success: false,
        error: {
          code: 'INCIDENT_RESPONSE_FAILED',
          message: (error as Error).message,
          recoverable: true,
        },
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    }
  }

  async terminate(): Promise<void> {
    await this.playbookExecutor.terminate();
    await this.alertCorrelator.terminate();
    await this.evidenceCollector.terminate();
    await this.reportGenerator.terminate();
  }

  static getDefinition(): HandDefinition {
    return {
      id: 'incident-responder',
      name: 'Incident Responder Hand',
      description: 'Automated security incident response',
      category: 'security',
      version: '1.0.0',
      requirements: [],
      settings: [],
      metrics: [],
      tools: [],
      systemPrompt: '',
    };
  }
}
```

import yaml from 'js-yaml';
import type { Playbook, PlaybookStep, PlaybookResult } from './types';

export class PlaybookExecutor {
  private playbooks: Map<string, Playbook> = new Map();

  async initialize(): Promise<void> {
    // 加载默认剧本
    await this.loadDefaultPlaybooks();
  }

  async loadPlaybook(content: string, format: 'yaml' | 'json'): Promise<Playbook> {
    const playbook = format === 'yaml' 
      ? yaml.load(content) as Playbook
      : JSON.parse(content) as Playbook;
    
    this.playbooks.set(playbook.id, playbook);
    return playbook;
  }

  async execute(
    playbookId: string,
    context: Record<string, unknown>
  ): Promise<PlaybookResult> {
    const playbook = this.playbooks.get(playbookId);
    if (!playbook) {
      throw new Error(`Playbook not found: ${playbookId}`);
    }

    const results: PlaybookStep[] = [];

    for (const step of playbook.steps) {
      // 检查条件
      if (step.condition && !this.evaluateCondition(step.condition, context)) {
        continue;
      }

      // 执行步骤
      const result = await this.executeStep(step, context);
      results.push(result);

      // 失败处理
      if (!result.success && step.critical) {
        await this.handleFailure(step, result);
        break;
      }
    }

    return {
      playbookId,
      success: results.every(r => r.success),
      steps: results,
      timestamp: new Date(),
    };
  }

  private async executeStep(
    step: PlaybookStep,
    context: Record<string, unknown>
  ): Promise<PlaybookStep> {

  }
