## 示例代码片段

### scheduler.ts 骨架

```typescript
import { Cron } from "croner";
import { PriorityTaskQueue } from "./queue.js";
import { HandExecutor } from "./executor.js";
import type { ScheduleConfig, SchedulePolicy, SchedulerConfig } from "./types.js";
import { HandRegistry } from "../registry.js";

export class HandScheduler {
  private queue: PriorityTaskQueue;
  private executor: HandExecutor;
  private schedules: Map<string, ScheduleConfig> = new Map();
  private cronJobs: Map<string, Cron> = new Map();
  private tickInterval?: NodeJS.Timeout;
  private running = false;
  
  constructor(
    private config: SchedulerConfig,
    private registry: HandRegistry
  ) {
    this.queue = new PriorityTaskQueue();
    this.executor = new HandExecutor({
      maxConcurrency: config.maxConcurrency,
      defaultTimeout: config.defaultTimeout,
    });
  }
  
  register(handId: string, policy: SchedulePolicy): void {
    const config: ScheduleConfig = {
      handId,
      enabled: true,
      policy,
      runCount: 0,
      errorCount: 0,
    };
    
    this.schedules.set(handId, config);
    
    // 设置cron任务
    if (policy.cron) {
      const job = new Cron(policy.cron, () => this.enqueueTask(handId));
      this.cronJobs.set(handId, job);
    }
    
    // 设置间隔任务
    if (policy.interval) {
      const job = new Cron(`*/${Math.floor(policy.interval / 60000)} * * * *`, () => {
        this.enqueueTask(handId);
      });
      this.cronJobs.set(handId, job);
    }
  }
  
  unregister(handId: string): void {
    this.schedules.delete(handId);
    this.queue.remove(handId);
    
    const job = this.cronJobs.get(handId);
    if (job) {
      job.stop();
      this.cronJobs.delete(handId);
    }
  }
  
  async start(): Promise<void> {
    this.running = true;
    this.tickInterval = setInterval(() => this.tick(), this.config.tickInterval);
  }
  
  async stop(): Promise<void> {
    this.running = false;
    if (this.tickInterval) {
      clearInterval(this.tickInterval);
    }
    for (const job of this.cronJobs.values()) {
      job.stop();
    }
    this.cronJobs.clear();
  }
  
  private async tick(): Promise<void> {
    while (!this.queue.isEmpty() && this.executor.getAvailableSlots() > 0) {
      const task = this.queue.dequeue();
      if (task) {
        this.executor.execute(task).catch((err) => {
          this.handleExecutionError(task, err);
        });
      }
    }
  }
  
  async runNow(handId: string): Promise<HandResult> {
    const config = this.schedules.get(handId);
    if (!config) {
      throw new Error(`Hand ${handId} not registered`);
    }
    
    const instance = await this.registry.activate(handId, {});
    return this.executor.execute({
      handId,
      instanceId: instance.instanceId,
      scheduledTime: new Date(),
      priority: config.policy.priority,
      retryCount: 0,
    });
  }
  
  private enqueueTask(handId: string): void {
    const config = this.schedules.get(handId);
    if (!config || !config.enabled) return;
    
    this.queue.enqueue({
      handId,
      instanceId: crypto.randomUUID(),
      scheduledTime: new Date(),
      priority: config.policy.priority,
      retryCount: 0,
    });
    
    config.nextRun = this.calculateNextRun(config.policy);
  }
  
  private async handleExecutionError(task: ScheduledTask, error: Error): Promise<void> {
    const config = this.schedules.get(task.handId);
    if (!config) return;
    
    config.errorCount++;
    
    if (task.retryCount < config.policy.retry.maxAttempts) {
      const delay = calculateBackoff(config.policy.retry, task.retryCount);
      setTimeout(() => {
        this.queue.enqueue({
          ...task,
          retryCount: task.retryCount + 1,
          scheduledTime: new Date(Date.now() + delay),
        });
      }, delay);
    }
  }
}
```

## 调度策略

```typescript
interface SchedulePolicy {
  // 定时策略
  cron?: string;           // "0 6 * * *" - 每天6点
  interval?: number;       // 3600000 - 每小时
  delay?: number;          // 60000 - 延迟1分钟
  
  // 并发策略
  concurrency: number;     // 最大并发数
  rateLimit?: number;      // 每分钟请求数
  
  // 重试策略
  retry: {
    maxAttempts: number;   // 3
    backoff: 'fixed' | 'exponential' | 'linear';
    baseDelay: number;     // 1000ms
    maxDelay: number;      // 60000ms
  };
  
  // 优先级
  priority: 'low' | 'normal' | 'high' | 'critical';
}
```

---

## 注意事项

1. **时区**: cron表达式使用UTC时区
2. **并发**: 必须控制最大并发数，避免资源耗尽
3. **重试**: 指数退避最大延迟不应超过配置的maxDelay
4. **持久化**: 调度状态需要持久化，重启后恢复
5. **优雅关闭**: 停止时等待正在执行的任务完成
