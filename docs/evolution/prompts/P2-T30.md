### detector.ts 骨架

```typescript
import { SHA256Hasher } from './hasher';
import { CallTracker } from './tracker';
import { ToolCall, LoopDetectionResult } from './types';

export class LoopDetector {
  private hasher: SHA256Hasher;
  private tracker: CallTracker;
  private callFingerprints: Map<string, number>;
  private config: LoopProtectionConfig;

  constructor(config?: Partial<LoopProtectionConfig>) {
    this.hasher = new SHA256Hasher();
    this.tracker = new CallTracker();
    this.callFingerprints = new Map();
    this.config = { ...DEFAULT_CONFIG, ...config };
  }

  /**
   * 检测循环
   */
  detectLoop(call: ToolCall): LoopDetectionResult {
    // 计算指纹
    const fingerprint = this.calculateFingerprint(call);
    
    // 追踪调用
    this.tracker.trackCall(call, fingerprint);
    
    // 检查重复
    const count = this.callFingerprints.get(fingerprint) || 0;
    this.callFingerprints.set(fingerprint, count + 1);
    
    // 判断是否循环
    const isLoop = count >= this.config.threshold;
    
    // 检查乒乓模式
    const isPingPong = this.handlePingPongPattern(this.tracker.getRecentCalls(10));
    
    return {
      isLoop,
      isPingPong,
      fingerprint,
      count: count + 1,
      threshold: this.config.threshold,
      severity: this.calculateSeverity(count + 1),
    };
  }

  /**
   * 计算指纹
   */
  calculateFingerprint(call: ToolCall): string {
    // 序列化调用（工具名 + 参数）
    const serialized = JSON.stringify({
      tool: call.tool,
      args: call.args,
    });
    
    // 计算SHA256哈希
    return this.hasher.hash(serialized);
  }

  /**
   * 检查重复
   */
  checkDuplicate(fingerprint: string): boolean {
    const count = this.callFingerprints.get(fingerprint) || 0;
    return count > 0;
  }

  /**
   * 处理乒乓模式
   */
  handlePingPongPattern(calls: ToolCall[]): boolean {
    if (calls.length < 4) return false;

    // 检测A-B-A-B模式
    for (let i = 0; i < calls.length - 3; i++) {
      const a1 = this.calculateFingerprint(calls[i]);
      const b1 = this.calculateFingerprint(calls[i + 1]);
      const a2 = this.calculateFingerprint(calls[i + 2]);
      const b2 = this.calculateFingerprint(calls[i + 3]);

      if (a1 === a2 && b1 === b2 && a1 !== b1) {
        return true;
      }
    }

    return false;
  }

  /**
   * 计算严重性
   */
  private calculateSeverity(count: number): 'low' | 'medium' | 'high' | 'critical' {
    const ratio = count / this.config.threshold;
    
    if (ratio >= 2) return 'critical';
    if (ratio >= 1.5) return 'high';
    if (ratio >= 1) return 'medium';
    return 'low';
  }

  /**
   * 清理过期记录
   */
  cleanup(): void {
    // 清理超过时间窗口的记录
    const cutoff = Date.now() - this.config.windowMs;
    
    for (const [fingerprint, count] of this.callFingerprints.entries()) {
      // 如果在时间窗口外，移除记录
      // 实际实现需要记录时间戳

  }
}
```

### hasher.ts 骨架

```typescript
import crypto from 'crypto';

export class SHA256Hasher {
  /**
   * 计算SHA256哈希
   */
  hash(data: string): string {
    return crypto.createHash('sha256').update(data).digest('hex');
  }

  /**
   * 计算对象哈希
   */
  hashObject(obj: any): string {
    const serialized = JSON.stringify(obj, Object.keys(obj).sort());
    return this.hash(serialized);
  }

  /**
   * 计算调用哈希
   */
  hashCall(tool: string, args: any): string {
    return this.hashObject({ tool, args });
  }
}
```

### circuit-breaker.ts 骨架

```typescript
import { CircuitBreakerConfig, CircuitBreakerState } from './types';

export class CircuitBreaker {
  private state: CircuitBreakerState = 'closed';
  private failureCount: number = 0;
  private lastFailureTime: number = 0;
  private config: CircuitBreakerConfig;

  constructor(config: CircuitBreakerConfig) {
    this.config = config;
  }

  /**
   * 执行操作（带熔断）
   */
  async execute<T>(fn: () => Promise<T>): Promise<T> {
    // 检查熔断状态
    if (this.state === 'open') {
      // 检查是否可以尝试半开
      if (this.shouldAttemptReset()) {
        this.state = 'half-open';
      } else {
        throw new Error('Circuit breaker is open');
      }
    }

    try {
      const result = await fn();
      
      // 成功，重置
      if (this.state === 'half-open') {
        this.reset();
      }
      
      return result;
    } catch (error) {
      // 失败，记录
      this.recordFailure();
      throw error;
    }
  }

  /**
   * 检查熔断状态
   */
  isOpen(): boolean {
    return this.state === 'open';
  }

  /**
   * 触发熔断
   */
  trip(): void {
    this.state = 'open';
    this.lastFailureTime = Date.now();
    
    // 调用回调
    if (this.config.onBreak) {
      this.config.onBreak();
    }
  }

  /**
   * 重置熔断器
   */
  reset(): void {
    this.state = 'closed';
    this.failureCount = 0;
    this.lastFailureTime = 0;
  }

  /**
   * 记录失败
   */
  private recordFailure(): void {
    this.failureCount++;
    this.lastFailureTime = Date.now();

    if (this.failureCount >= this.config.maxCalls) {
      this.trip();
    }
  }

  /**
   * 是否应该尝试重置
   */
  private shouldAttemptReset(): boolean {
    const elapsed = Date.now() - this.lastFailureTime;
    return elapsed >= this.config.cooldownMs;
  }

  /**
   * 获取状态
   */
  getState(): CircuitBreakerState {
    return this.state;
  }

  /**
   * 获取失败计数
   */
  getFailureCount(): number {
    return this.failureCount;
  }
}
```

### tracker.ts 骨架

```typescript
import { ToolCall } from './types';

export class CallTracker {
  private calls: Array<{
    call: ToolCall;
    fingerprint: string;
    timestamp: number;
  }>;

  constructor() {
    this.calls = [];
  }

  /**
   * 追踪调用
   */
  trackCall(call: ToolCall, fingerprint: string): void {
    this.calls.push({
      call,
      fingerprint,
      timestamp: Date.now(),
    });

    // 限制历史大小
    if (this.calls.length > 1000) {
      this.calls.shift();
    }
  }

  /**
   * 获取调用历史
   */
  getCallHistory(): Array<{
    call: ToolCall;
    fingerprint: string;
    timestamp: number;
  }> {
    return [...this.calls];
  }

  /**
   * 获取最近N次调用
   */
  getRecentCalls(n: number): ToolCall[] {
    return this.calls.slice(-n).map(c => c.call);
  }

  /**
   * 清除历史
   */
  clearHistory(): void {
    this.calls = [];
  }

  /**
   * 分析模式
   */
  analyzePatterns(): {
    totalCalls: number;
    uniqueCalls: number;
    topCalls: Array<{ fingerprint: string; count: number }>;
  } {
    const fingerprintCounts = new Map<string, number>();

    for (const { fingerprint } of this.calls) {
      fingerprintCounts.set(fingerprint, (fingerprintCounts.get(fingerprint) || 0) + 1);
    }

    const topCalls = Array.from(fingerprintCounts.entries())
      .map(([fingerprint, count]) => ({ fingerprint, count }))
      .sort((a, b) => b.count - a.count)
      .slice(0, 10);

    return {
      totalCalls: this.calls.length,
      uniqueCalls: fingerprintCounts.size,
      topCalls,
    };
  }
}
```
