## 示例代码片段

### bot-adapter.ts 骨架

```typescript
import { CloudAdapter, ConfigurationBotFrameworkAuthentication } from 'botbuilder';
import { BaseChannel, ChannelConfig, ChannelStatus } from '../base';
import type { MSTeamsConfig } from './types';

export class MSTeamsChannel extends BaseChannel {
  private adapter: CloudAdapter;
  
  constructor(config: MSTeamsConfig) {
    super(config);
    
    const auth = new ConfigurationBotFrameworkAuthentication({
      MicrosoftAppId: config.appId,
      MicrosoftAppPassword: config.appPassword,
    });
    
    this.adapter = new CloudAdapter(auth);
  }

  async processActivity(req: Request, res: Response, logic: (context: TurnContext) => Promise<void>): Promise<void> {
    await this.adapter.process(req, res, logic);
  }

  async sendProactiveMessage(conversationReference: ConversationReference, message: string): Promise<void> {
    await this.adapter.continueConversation(conversationReference, async (context) => {
      await context.sendActivity(message);
    });
  }
}
```
