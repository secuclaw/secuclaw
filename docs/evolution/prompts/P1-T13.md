## 示例代码片段

### server.ts 骨架

```typescript
import http from "node:http";
import type { Socket } from "node:net";
import { WebSocketServer, type WebSocket } from "ws";
import { authorizeGatewayConnect } from "./auth.js";
import { SessionManager } from "./session-manager.js";

export type GatewayServerOpts = {
  port?: number;
  bind?: "loopback" | "all";
  auth?: GatewayAuthConfig;
  runtime: RuntimeEnv;
};

export class GatewayServer {
  private httpServer: http.Server;
  private wss: WebSocketServer;
  private connections = new Set<WebSocket>();
  private methods = new Map<string, MethodHandler>();
  private sessionManager: SessionManager;
  
  constructor(private opts: GatewayServerOpts) {
    this.httpServer = http.createServer();
    this.wss = new WebSocketServer({ noServer: true });
    this.sessionManager = new SessionManager();
    this.registerDefaultMethods();
  }
  
  async start(): Promise<void> {
    this.wss.on("connection", (ws, req) => this.handleConnection(ws, req));
    
    this.httpServer.on("upgrade", (req, socket, head) => {
      this.wss.handleUpgrade(req, socket as Socket, head, (ws) => {
        this.wss.emit("connection", ws, req);
      });
    });
    
    const host = this.opts.bind === "all" ? "0.0.0.0" : "127.0.0.1";
    await new Promise<void>((resolve) => this.httpServer.listen(this.opts.port ?? 18789, host, resolve));
  }
  
  private async handleConnection(ws: WebSocket, req: IncomingMessage): Promise<void> {
    const authResult = await authorizeGatewayConnect({
      auth: this.opts.auth,
      req,
    });
    
    if (!authResult.ok) {
      ws.close(1008, authResult.reason);
      return;
    }
    
    this.connections.add(ws);
    ws.on("message", (data) => this.handleMessage(ws, data));
    ws.on("close", () => this.connections.delete(ws));
  }
  
  private async handleMessage(ws: WebSocket, data: Buffer): Promise<void> {
    const frame = JSON.parse(data.toString()) as GatewayFrame;
    if (frame.type !== "request") return;
    
    const handler = this.methods.get(frame.method);
    if (!handler) {
      this.sendResponse(ws, frame.id, undefined, { code: -32601, message: "Method not found" });
      return;
    }
    
    try {
      const result = await handler(frame.params, { connection: ws });
      this.sendResponse(ws, frame.id, result);
    } catch (err) {
      this.sendResponse(ws, frame.id, undefined, { code: -32603, message: String(err) });
    }
  }
  
  registerMethod(name: string, handler: MethodHandler): void {
    this.methods.set(name, handler);
  }
  
  broadcast(message: GatewayMessage): void {
    const data = JSON.stringify(message);
    for (const ws of this.connections) {
      ws.send(data);
    }
  }
  
  async stop(): Promise<void> {
    for (const ws of this.connections) {
      ws.close(1001, "Server shutting down");
    }
    this.wss.close();
    this.httpServer.close();
  }
}
```

### protocol.ts 帧

```typescript
export type GatewayFrame = 
  | { type: 'request'; id: string; method: string; params: unknown }
  | { type: 'response'; id: string; result?: unknown; error?: GatewayError }
  | { type: 'event'; event: string; data: unknown };

export interface GatewayError {
  code: number;
  message: string;
  data?: unknown;
}

export function createRequest(id: string, method: string, params: unknown): GatewayFrame {
  return { type: 'request', id, method, params };
}

export function createResponse(id: string, result?: unknown, error?: GatewayError): GatewayFrame {
  return { type: 'response', id, result, error };
}

export function createEvent(event: string, data: unknown): GatewayFrame {
  return { type: 'event', event, data };
}
```

## 注意事项

1. **会话隔离**: 不同客户端的会话必须完全隔离
2. **认证安全**: 所有连接必须经过认证
3. **优雅关闭**: 关闭时必须等待所有请求完成
4. **错误处理**: 方法调用失败必须返回标准错误格式
5. **限流保护**: 认证失败需要限流保护
