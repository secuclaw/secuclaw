## 示例代码片段

### tracker.ts 骨架

```typescript
import { TaintLabel, TaintSource, TaintSink, TaintedData, PolicyResult } from './types';
import { TaintSourceRegistry } from './sources';
import { TaintSinkRegistry } from './sinks';
import { PropagationEngine } from './propagation';
import { TaintPolicyManager } from './policy';
import { LabelManager } from './labels';

export class TaintTracker {
  private sourceRegistry: TaintSourceRegistry;
  private sinkRegistry: TaintSinkRegistry;
  private propagationEngine: PropagationEngine;
  private policyManager: TaintPolicyManager;
  private labelManager: LabelManager;
  
  // 使用WeakMap存储污点信息，避免内存泄漏
  private taintMap: WeakMap<object, TaintLabel[]>;

  constructor() {
    this.sourceRegistry = new TaintSourceRegistry();
    this.sinkRegistry = new TaintSinkRegistry();
    this.propagationEngine = new PropagationEngine();
    this.policyManager = new TaintPolicyManager();
    this.labelManager = new LabelManager();
    this.taintMap = new WeakMap();
  }

  /**
   * 标记数据源
   */
  taintSource(source: TaintSource, data: any): TaintedData {
    const labels = this.sourceRegistry.getLabels(source);
    
    // 创建代理对象追踪数据
    const proxy = this.createTaintProxy(data, labels);
    
    // 存储污点信息
    if (typeof data === 'object' && data !== null) {
      this.taintMap.set(data, labels);
    }

    return {
      data: proxy,
      labels,
      source,
      timestamp: new Date(),
    };
  }

  /**
   * 传播污点
   */
  propagate(tainted: TaintedData, operation: string): TaintedData {
    // 应用传播规则
    const propagatedLabels = this.propagationEngine.propagate(
      tainted.labels,
      operation
    );

    // 创建新的污点数据
    return {
      ...tainted,
      labels: propagatedLabels,
      operation,
      timestamp: new Date(),
    };
  }

  /**
   * 检查数据汇
   */
  checkSink(sink: TaintSink, data: any): PolicyResult {
    // 获取数据的污点标签
    const labels = this.getLabels(data);
    
    // 检查策略
    const policy = this.policyManager.getPolicy(sink, labels);
    
    // 执行策略检查
    return this.policyManager.enforcePolicy(policy, {
      sink,
      labels,
      data,
    });
  }

  /**
   * 添加标签
   */
  addLabel(data: any, label: TaintLabel): void {
    if (typeof data === 'object' && data !== null) {
      const existingLabels = this.taintMap.get(data) || [];
      const newLabels = this.labelManager.mergeLabels(existingLabels, [label]);
      this.taintMap.set(data, newLabels);
    }
  }

  /**
   * 获取标签
   */
  getLabels(data: any): TaintLabel[] {
    if (typeof data === 'object' && data !== null) {
      return this.taintMap.get(data) || [];
    }
    return [];
  }

  /**
   * 分析数据流
   */
  analyzeFlow(source: TaintSource, sink: TaintSink): FlowPath[] {
    // 构建流图
    const flowGraph = this.buildFlowGraph();
    
    // 查找从源到汇的路径
    const paths = this.findPaths(flowGraph, source, sink);
    
    return paths;
  }

  /**
   * 创建污点代理
   */
  private createTaintProxy(data: any, labels: TaintLabel[]): any {
    if (typeof data !== 'object' || data === null) {
      return data;
    }

    return new Proxy(data, {
      get: (target, property, receiver) => {
        const value = Reflect.get(target, property, receiver);
        
        // 传播污点到属性
        if (typeof value === 'object' && value !== null) {
          this.addLabel(value, labels[0]);
        }
        
        return value;
      },
      
      set: (target, property, value, receiver) => {
        // 传播污点
        if (typeof value === 'object' && value !== null) {
          this.addLabel(value, labels[0]);
        }
        
        return Reflect.set(target, property, value, receiver);
      },
    });
  }
}
```

### propagation.ts 骨架

```typescript
import { TaintLabel, PropagationRule } from './types';

export class PropagationEngine {
  private rules: Map<string, PropagationRule>;

  constructor() {
    this.rules = new Map();
    this.initializeDefaultRules();
  }

  /**
   * 传播污点
   */
  propagate(labels: TaintLabel[], operation: string): TaintLabel[] {
    const rule = this.rules.get(operation);
    
    if (!rule) {
      // 默认：保守传播（保持所有标签）
      return labels;
    }

    return rule.apply(labels);
  }

  /**
   * 应用传播规则
   */
  applyRules(labels: TaintLabel[], operations: string[]): TaintLabel[] {
    let result = labels;
    
    for (const operation of operations) {
      result = this.propagate(result, operation);
    }
    
    return result;
  }

  /**
   * 合并标签
   */
  combineLabels(labelSets: TaintLabel[][]): TaintLabel[] {
    const combined = new Set<TaintLabel>();
    
    for (const labels of labelSets) {
      for (const label of labels) {
        combined.add(label);
      }
    }
    
    return Array.from(combined);
  }

  /**
   * 清除污点
   */
  sanitize(labels: TaintLabel[], sanitizer: string): TaintLabel[] {
    // 根据sanitizer类型清除特定标签
    switch (sanitizer) {
      case 'html_escape':
        return labels.filter(l => l !== 'xss');
      case 'sql_escape':
        return labels.filter(l => l !== 'sql_injection');
      case 'path_canonicalize':
        return labels.filter(l => l !== 'path_traversal');
      default:
        return labels;
    }
  }

  /**
   * 初始化默认传播规则
   */
  private initializeDefaultRules(): void {
    // 赋值操作：完全传播
    this.rules.set('assignment', {
      apply: (labels) => labels,
    });

    // 字符串拼接：完全传播
    this.rules.set('concatenation', {
      apply: (labels) => labels,
    });

    // 算术运算：降级传播
    this.rules.set('arithmetic', {
      apply: (labels) => labels.filter(l => l !== 'pii'),
    });

    // 函数调用：保守传播
    this.rules.set('function_call', {
      apply: (labels) => labels,
    });

    // 加密操作：清除机密标签
    this.rules.set('encryption', {
      apply: (labels) => labels.filter(l => l !== 'confidential'),
    });

    // 哈希操作：清除所有标签
    this.rules.set('hashing', {
      apply: () => [],
    });
  }
}
```

### types.ts 骨架

```typescript
// 污点标签类型
export type TaintLabel =
  | 'confidential'    // 机密
  | 'sensitive'       // 敏感
  | 'pii'             // 个人身份信息
  | 'secret'          // 秘密
  | 'public'          // 公开
  | 'xss'             // XSS风险
  | 'sql_injection'   // SQL注入风险
  | 'path_traversal'; // 路径遍历风险

// 污点源类型
export type TaintSourceType =
  | 'user_input'      // 用户输入
  | 'file_read'       // 文件读取
  | 'network_recv'    // 网络接收
  | 'database_query'  // 数据库查询
  | 'env_variable';   // 环境变量

// 污点汇类型
export type TaintSinkType =
  | 'network_send'    // 网络发送
  | 'file_write'      // 文件写入
  | 'database_write'  // 数据库写入
  | 'command_exec'    // 命令执行
  | 'log_output';     // 日志输出

// 污点源
export interface TaintSource {
  id: string;
  type: TaintSourceType;
  labels: TaintLabel[];
  description: string;
  metadata?: Record<string, unknown>;
}

// 污点汇
export interface TaintSink {
  id: string;
  type: TaintSinkType;
  allowedLabels: TaintLabel[];
  deniedLabels: TaintLabel[];
  description: string;
  metadata?: Record<string, unknown>;
}

// 污染数据
export interface TaintedData {
  data: any;
  labels: TaintLabel[];
  source: TaintSource;
  operation?: string;
  timestamp: Date;
}

// 策略类型
export type PolicyType = 'allow' | 'deny' | 'sanitize' | 'audit';

// 污点策略
export interface TaintPolicy {
  id: string;
  type: PolicyType;
  source: TaintSource;
  sink: TaintSink;
  labels: TaintLabel[];
  action: () => void;
  description: string;
}

// 策略检查结果
export interface PolicyResult {
  allowed: boolean;
  policy: TaintPolicy;
  violations: PolicyViolation[];
  sanitized: boolean;
}

// 策略违规
export interface PolicyViolation {
  policy: TaintPolicy;
  labels: TaintLabel[];
  data: any;
  timestamp: Date;
  stackTrace?: string;
}

// 传播规则
export interface PropagationRule {
  apply: (labels: TaintLabel[]) => TaintLabel[];
}

// 流路径
export interface FlowPath {
  source: TaintSource;
  sink: TaintSink;
  operations: string[];
  labels: TaintLabel[];
  risk: 'high' | 'medium' | 'low';
}
```
