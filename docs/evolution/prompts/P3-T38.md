## 示例代码片段

### client.ts 骨架

```typescript
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export class CloudflareClient {
  private connected: boolean = false;

  /**
   * 连接
   */
  async connect(): Promise<void> {
    try {
      await execAsync('cloudflared tunnel run');
      this.connected = true;
    } catch (error) {
      throw new Error(`Failed to connect: ${error}`);
    }
  }

  /**
   * 断开
   */
  async disconnect(): Promise<void> {
    try {
      await execAsync('pkill cloudflared');
      this.connected = false;
    } catch (error) {
      throw new Error(`Failed to disconnect: ${error}`);
    }
  }

  /**
   * 获取状态
   */
  async getStatus(): Promise<CloudflareStatus> {
    try {
      const { stdout } = await execAsync('cloudflared tunnel list');
      return { connected: this.connected, tunnels: stdout };
    } catch (error) {
      throw new Error(`Failed to get status: ${error}`);
    }
  }
}
```
