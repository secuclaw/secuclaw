## 任务描述

实现自主渗透测试Hand，模拟攻击链并生成渗透报告。该Hand负责自动化渗透测试流程，包括侦察、利用验证、后渗透阶段，并基于MITRE ATT&CK框架生成详细的渗透测试报告。

**⚠️ 重要提示**: 仅用于授权测试环境，未经授权的渗透测试是违法行为。

```typescript
import { BaseHand, HandContext, HandResult, HandDefinition } from '../base';
import { AttackChainSimulator } from './attack-chain-sim';
import { ReconnaissancePhase } from './reconnaissance';
import { ExploitationPhase } from './exploitation';
import { PostExploitationPhase } from './post-exploit';
import { MitreMapper } from './mitre-mapper';
import { PenTestReportGenerator } from './report-generator';
import type { Target, AttackChainResult, PenTestReport } from './types';

export class PenTesterHand extends BaseHand {
  private attackChainSim: AttackChainSimulator;
  private reconPhase: ReconnaissancePhase;
  private exploitPhase: ExploitationPhase;
  private postExploitPhase: PostExploitationPhase;
  private mitreMapper: MitreMapper;
  private reportGenerator: PenTestReportGenerator;

  constructor() {
    const definition: HandDefinition = {
      id: 'pen-tester',
      name: 'Penetration Tester Hand',
      description: 'Autonomous penetration testing with MITRE ATT&CK mapping',
      category: 'security',
      version: '1.0.0',
      requirements: [
        {
          key: 'authorization',
          label: 'Written Authorization',
          requirementType: 'env-var',
          checkValue: 'PENTEST_AUTHORIZED',
          description: 'Must have written authorization before testing',
        },
      ],
      settings: [
        {
          key: 'test_scope',
          label: 'Test Scope',
          description: 'Authorized targets and testing boundaries',
          settingType: 'textarea',
          default: '',
        },
        {
          key: 'test_intensity',
          label: 'Test Intensity',
          description: 'Testing intensity level',
          settingType: 'select',
          default: 'moderate',
          options: [
            { value: 'light', label: 'Light (Non-intrusive)' },
            { value: 'moderate', label: 'Moderate (Standard)' },
            { value: 'thorough', label: 'Thorough (Comprehensive)' },
          ],
        },
        {
          key: 'destructive_ops',
          label: 'Destructive Operations',
          description: 'Allow destructive operations (NOT RECOMMENDED)',
          settingType: 'toggle',
          default: 'false',
        },
      ],
      metrics: [
        { label: 'Tests Executed', memoryKey: 'tests_executed', format: 'number' },
        { label: 'Vulnerabilities Found', memoryKey: 'vulns_found', format: 'number' },
        { label: 'Attack Paths Tested', memoryKey: 'attack_paths', format: 'number' },
      ],
      tools: [
        'shell_exec',
        'web_fetch',
        'memory_store',
        'memory_recall',
        'knowledge_add_entity',
        'knowledge_query',
      ],
      systemPrompt: `You are an autonomous penetration tester. Your role is to:
1. Perform authorized security testing only
2. Simulate attack chains using Kill Chain and MITRE ATT&CK frameworks
3. Validate vulnerabilities without causing damage
4. Document all findings with evidence
5. Generate comprehensive penetration test reports

⚠️ CRITICAL: Only test systems you have explicit written authorization to test. Never perform destructive operations. Always document your activities.

Follow the rules of engagement:
- Stay within authorized scope
- Do not access production data
- Do not perform denial of service attacks
- Report all findings immediately
- Preserve evidence chain of custody`,
    };

    super(definition);
    this.attackChainSim = new AttackChainSimulator();
    this.reconPhase = new ReconnaissancePhase();
    this.exploitPhase = new ExploitationPhase();
    this.postExploitPhase = new PostExploitationPhase();
    this.mitreMapper = new MitreMapper();
    this.reportGenerator = new PenTestReportGenerator();
  }

  async initialize(): Promise<void> {
    // 验证授权
    const authorized = process.env.PENTEST_AUTHORIZED === 'true';
    if (!authorized) {
      throw new Error('Penetration testing requires explicit authorization. Set PENTEST_AUTHORIZED=true');
    }

    await this.attackChainSim.initialize();
    await this.reconPhase.initialize();
    await this.exploitPhase.initialize();
    await this.postExploitPhase.initialize();
    await this.mitreMapper.initialize();
    await this.reportGenerator.initialize();
  }

  async execute(context: HandContext): Promise<HandResult> {
    const startTime = Date.now();
    const testResults: any[] = [];

    try {
      // 1. 获取授权测试目标
      const targets = await this.getAuthorizedTargets(context);
      this.reportProgress(5, `Found ${targets.length} authorized targets`);

      // 2. 对每个目标进行测试
      for (let i = 0; i < targets.length; i++) {
        const target = targets[i];
        this.reportProgress(5 + (i / targets.length) * 70, `Testing target: ${target.name}`);

        // 2.1 侦察阶段
        this.reportProgress(10, 'Phase 1: Reconnaissance');
        const reconResult = await this.reconPhase.execute(target);

        // 2.2 攻击链模拟
        this.reportProgress(25, 'Phase 2: Attack Chain Simulation');
        const attackChain = await this.attackChainSim.simulate(target, reconResult);

        // 2.3 利用验证（非破坏性）
        this.reportProgress(40, 'Phase 3: Exploitation Validation');
        const exploitResults = await this.exploitPhase.execute(reconResult.vulnerabilities);

        // 2.4 后渗透测试（可选）
        if (this.shouldPerformPostExploit(exploitResults)) {
          this.reportProgress(55, 'Phase 4: Post-Exploitation');
          const postExploitResult = await this.postExploitPhase.execute(exploitResults);
          testResults.push({ target, reconResult, attackChain, exploitResults, postExploitResult });
        } else {
          testResults.push({ target, reconResult, attackChain, exploitResults });
        }

        // 2.5 MITRE映射
        this.reportProgress(65, 'Mapping to MITRE ATT&CK');
        const mitreMapping = await this.mitreMapper.mapResults(testResults[testResults.length - 1]);
        testResults[testResults.length - 1].mitreMapping = mitreMapping;
      }

      // 3. 生成报告
      this.reportProgress(85, 'Generating penetration test report');
      const report = await this.reportGenerator.generate(testResults);

      // 4. 更新指标
      this.updateMetrics(testResults);

      return {
        success: true,
        data: {
          report,
          targetsTested: targets.length,
          vulnerabilitiesFound: testResults.reduce((sum, r) => sum + r.exploitResults.length, 0),
        },
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    } catch (error) {
      await this.handleError(error as Error);
      return {
        success: false,
        error: {
          code: 'PEN_TEST_FAILED',
          message: (error as Error).message,
          recoverable: false,
        },
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    }
  }

  async terminate(): Promise<void> {
    // 清理资源和证据
  }

  private async getAuthorizedTargets(context: HandContext): Promise<Target[]> {
    // 从配置中获取授权目标列表
    const scope = this.getConfig().test_scope;
    if (!scope) {
      throw new Error('No authorized test scope defined');
    }
    // 解析并返回目标
    return [];
  }

  private shouldPerformPostExploit(exploitResults: any[]): boolean {
    // 根据配置和利用结果决定是否执行后渗透测试
    return this.getConfig().test_intensity === 'thorough' && exploitResults.length > 0;
  }

  private updateMetrics(testResults: any[]): void {
    const totalTests = testResults.length;
    const totalVulns = testResults.reduce((sum, r) => sum + r.exploitResults.length, 0);
    const totalPaths = testResults.reduce((sum, r) => sum + r.attackChain.paths.length, 0);

    this.context.reportMetric('tests_executed', totalTests);
    this.context.reportMetric('vulns_found', totalVulns);
    this.context.reportMetric('attack_paths', totalPaths);
  }

  static getDefinition(): HandDefinition {
    return {
      id: 'pen-tester',
      name: 'Penetration Tester Hand',
      description: 'Autonomous penetration testing',
      category: 'security',
      version: '1.0.0',
      requirements: [],
      settings: [],
      metrics: [],
      tools: [],
      systemPrompt: '',
    };
  }
}
```

```typescript
import type { MitreTechnique, MitreTactic, MitreMapping } from './types';

export class MitreMapper {
  private techniques: Map<string, MitreTechnique>;
  private tactics: Map<string, MitreTactic>;

  async initialize(): Promise<void> {
    // 加载MITRE ATT&CK数据
    await this.loadMitreData();
  }

  /**
   * 映射测试结果到MITRE技术
   */
  async mapResults(testResult: any): Promise<MitreMapping> {
    const techniques: MitreTechnique[] = [];
    const tactics: MitreTactic[] = [];

    // 映射侦察阶段
    if (testResult.reconResult) {
      techniques.push(...this.mapReconToTechniques(testResult.reconResult));
    }

    // 映射利用阶段
    if (testResult.exploitResults) {
      techniques.push(...this.mapExploitToTechniques(testResult.exploitResults));
    }

    // 映射后渗透阶段
    if (testResult.postExploitResult) {
      techniques.push(...this.mapPostExploitToTechniques(testResult.postExploitResult));
    }

    // 去重并映射到战术
    const uniqueTechniques = this.deduplicateTechniques(techniques);
    const uniqueTactics = this.mapTechniquesToTactics(uniqueTechniques);

    return {
      techniques: uniqueTechniques,
      tactics: uniqueTactics,
      heatMap: this.generateHeatMap(uniqueTechniques),
    };
  }

  private mapReconToTechniques(reconResult: any): MitreTechnique[] {
    const techniques: MitreTechnique[] = [];

    // 被动侦察 -> T1591 (Gather Victim Organization Information)
    if (reconResult.passiveRecon) {
      techniques.push({
        id: 'T1591',
        name: 'Gather Victim Organization Information',
        tactic: 'Reconnaissance',
        description: 'Passive information gathering',
        detected: true,
      });
    }

    // 主动扫描 -> T1595 (Active Scanning)
    if (reconResult.activeRecon) {
      techniques.push({
        id: 'T1595',
        name: 'Active Scanning',
        tactic: 'Reconnaissance',
        description: 'Active port and service scanning',
        detected: true,
      });
    }

    return techniques;
  }

  private mapExploitToTechniques(exploitResults: any[]): MitreTechnique[] {
    const techniques: MitreTechnique[] = [];

    for (const exploit of exploitResults) {
      // 根据漏洞类型映射到MITRE技术
      if (exploit.type === 'web') {
        techniques.push({
          id: 'T1190',
          name: 'Exploit Public-Facing Application',
          tactic: 'Initial Access',
          description: 'Web application exploitation',
          detected: true,
        });
      }

      if (exploit.type === 'phishing') {
        techniques.push({
          id: 'T1566',
          name: 'Phishing',
          tactic: 'Initial Access',
          description: 'Phishing attack simulation',
          detected: true,
        });
      }
    }

    return techniques;
  }

  private mapPostExploitToTechniques(postExploitResult: any): MitreTechnique[] {
    const techniques: MitreTechnique[] = [];

    // 权限提升
    if (postExploitResult.privilegeEscalation) {
      techniques.push({
        id: 'T1068',
        name: 'Exploitation for Privilege Escalation',
        tactic: 'Privilege Escalation',
        description: 'Privilege escalation testing',
        detected: true,
      });
    }

    // 横向移动
    if (postExploitResult.lateralMovement) {
      techniques.push({
        id: 'T1021',
        name: 'Remote Services',
        tactic: 'Lateral Movement',
        description: 'Lateral movement testing',
        detected: true,
      });
    }

    return techniques;
  }

  private generateHeatMap(techniques: MitreTechnique[]): any {
    // 生成MITRE ATT&CK热力图
    const heatMap: any = {};

    for (const technique of techniques) {
      if (!heatMap[technique.tactic]) {
        heatMap[technique.tactic] = [];
      }
      heatMap[technique.tactic].push(technique);
    }

    return heatMap;
  }

  private deduplicateTechniques(techniques: MitreTechnique[]): MitreTechnique[] {
    const seen = new Set<string>();
    return techniques.filter(t => {
      if (seen.has(t.id)) return false;
      seen.add(t.id);
      return true;
    });
  }

  private mapTechniquesToTactics(techniques: MitreTechnique[]): MitreTactic[] {
    const tacticMap = new Map<string, MitreTactic>();

    for (const technique of techniques) {
      if (!tacticMap.has(technique.tactic)) {
        tacticMap.set(technique.tactic, {
          name: technique.tactic,
          techniques: [],
        });
      }
      tacticMap.get(technique.tactic)!.techniques.push(technique);
    }

    return Array.from(tacticMap.values());
  }

  private async loadMitreData(): Promise<void> {
    // 从 secuclaw/packages/core/data/mitre/ 加载数据
  }
}
```

```typescript
  phases: AttackPhase[];
  currentPhase: number;
  progress: number;
}

export interface AttackPhase {
  name: string;
  killChainPhase: string;
  mitreTactic: string;
  techniques: string[];
  status: 'pending' | 'in-progress' | 'completed' | 'failed';
  findings: Finding[];
}

export interface Vulnerability {
  id: string;
  cve?: string;
  name: string;
  severity: 'critical' | 'high' | 'medium' | 'low';
  cvss?: number;
  description: string;
  host: string;
  port?: number;
  service?: string;
  evidence: string[];
}

export interface MitreTechnique {
  id: string;
  name: string;
  tactic: string;
  description: string;
  detected: boolean;
  mitigations?: string[];
}

export interface MitreTactic {
  name: string;
  techniques: MitreTechnique[];
}

export interface MitreMapping {
  techniques: MitreTechnique[];
  tactics: MitreTactic[];
  heatMap: Record<string, MitreTechnique[]>;
}

export interface PenTestReport {
  id: string;
  title: string;
  executiveSummary: string;
  scope: Target[];
  findings: Finding[];
  mitreMapping: MitreMapping;
  recommendations: string[];
  generatedAt: Date;
}

export interface Finding {
  id: string;
  title: string;
  severity: 'critical' | 'high' | 'medium' | 'low';
  description: string;
  impact: string;
  recommendation: string;
  evidence: string[];
  mitreTechniques: string[];
}
```

## 攻击链阶段

```
1. 侦察 (Reconnaissance)
   └─ 被动/主动信息收集

2. 武器化 (Weaponization)
   └─ 准备攻击载荷

3. 投递 (Delivery)
   └─ 钓鱼/漏洞利用

4. 利用 (Exploitation)
   └─ 执行攻击代码

5. 安装 (Installation)
   └─ 持久化

6. 命令控制 (C2)
   └─ 建立通信

7. 目标达成 (Actions)
   └─ 数据窃取/破坏
```

## 注意事项

1. **授权验证**: 必须在初始化时验证书面授权
2. **范围限制**: 严格限制在授权范围内测试
3. **非破坏性**: 默认不执行破坏性操作
4. **证据保护**: 保护所有测试证据的完整性
5. **报告及时**: 及时生成和提交测试报告
6. **法律合规**: 遵守当地法律法规
