## 示例代码片段

### base.ts 骨架

```typescript
import { v4 as uuidv4 } from "uuid";
import type { 
  HandDefinition, 
  HandState, 
  HandContext, 
  HandResult, 
  HandError,
  ProgressCallback,
  ErrorCallback
} from "./types.js";

export abstract class BaseHand {
  protected definition: HandDefinition;
  protected state: HandState;
  protected progressCallbacks = new Set<ProgressCallback>();
  protected errorCallbacks = new Set<ErrorCallback>();
  
  constructor(definition: HandDefinition) {
    this.definition = definition;
    this.state = {
      instanceId: uuidv4(),
      status: "idle",
      progress: 0,
      runCount: 0,
      successCount: 0,
      errorCount: 0,
      metadata: {},
    };
  }
  
  // 生命周期 - 子类必须实现
  abstract initialize(): Promise<void>;
  abstract execute(context: HandContext): Promise<HandResult>;
  abstract terminate(): Promise<void>;
  
  // 状态管理
  getState(): HandState {
    return { ...this.state };
  }
  
  setState(updates: Partial<HandState>): void {
    this.state = { ...this.state, ...updates };
  }
  
  async persistState(): Promise<void> {
    // 保存到SQLite
  }
  
  // 进度报告
  onProgress(callback: ProgressCallback): () => void {
    this.progressCallbacks.add(callback);
    return () => this.progressCallbacks.delete(callback);
  }
  
  protected reportProgress(progress: number, message: string): void {
    this.state.progress = progress;
    for (const callback of this.progressCallbacks) {
      callback(progress, message);
    }
  }
  
  // 错误处理
  onError(callback: ErrorCallback): () => void {
    this.errorCallbacks.add(callback);
    return () => this.errorCallbacks.delete(callback);
  }
  
  protected async handleError(error: Error): Promise<void> {
    this.state.lastError = error;
    this.state.errorCount++;
    
    for (const callback of this.errorCallbacks) {
      callback(error);
    }
  }
  
  protected async retry<T>(
    fn: () => Promise<T>,
    attempts: number = 3,
    delay: number = 1000
  ): Promise<T> {
    let lastError: Error | undefined;
    
    for (let i = 0; i < attempts; i++) {
      try {
        return await fn();
      } catch (err) {
        lastError = err as Error;
        if (i < attempts - 1) {
          await new Promise((resolve) => setTimeout(resolve, delay * (i + 1)));
        }
      }
    }
    
    throw lastError;
  }
  
  // 指标
  getMetrics(): Record<string, unknown> {
    return {
      runCount: this.state.runCount,
      successCount: this.state.successCount,
      errorCount: this.state.errorCount,
      successRate: this.state.runCount > 0 
        ? this.state.successCount / this.state.runCount 
        : 0,
    };
  }
  
  // 静态方法
  static getDefinition(): HandDefinition {
    throw new Error("Subclass must implement getDefinition()");
  }
  
  static async checkRequirements(): Promise<RequirementCheckResult[]> {
    const definition = this.getDefinition();
    return RequirementChecker.checkAll(definition.requirements);
  }
}
```

## 注意事项

1. **生命周期**: 必须按顺序调用initialize → execute → terminate
2. **状态持久化**: 状态变化时自动持久化到SQLite
3. **错误恢复**: 可恢复错误自动重试，不可恢复错误通知上层
4. **资源清理**: terminate必须清理所有资源
5. **并发安全**: 状态更新需要加锁保护
