## 合规框架映射

```typescript
interface ComplianceFramework {
  id: string;
  name: string;
  version: string;
  domains: Domain[];
}

interface Domain {
  id: string;
  name: string;
  controls: Control[];
}

interface Control {
  id: string;
  description: string;
  checks: ComplianceCheck[];
  evidence: Evidence[];
  status: "compliant" | "partial" | "non-compliant" | "not-applicable";
}
```
