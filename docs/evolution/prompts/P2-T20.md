## 示例代码片段

### scanner-adapter.ts 骨架

```typescript
import { spawn } from "child_process";

export interface ScannerAdapter {
  name: string;
  scan(target: string, options: ScanOptions): Promise<ScanResult>;
  parseOutput(output: string): ParsedResult;
  isAvailable(): Promise<boolean>;
}

export class NmapAdapter implements ScannerAdapter {
  name = "nmap";
  
  async scan(target: string, options: ScanOptions): Promise<ScanResult> {
    const args = this.buildArgs(target, options);
    const output = await this.execute(args);
    return this.parseOutput(output);
  }
  
  private buildArgs(target: string, options: ScanOptions): string[] {
    const args = ["-oX", "-"];  // XML输出到stdout
    
    if (options.scanType === "quick") {
      args.push("-F");  // 快速扫描
    } else if (options.scanType === "stealth") {
      args.push("-sS");  // SYN扫描
    }
    
    args.push(target);
    return args;
  }
  
  private async execute(args: string[]): Promise<string> {
    return new Promise((resolve, reject) => {
      const process = spawn("nmap", args);
      let output = "";
      
      process.stdout.on("data", (data) => {
        output += data.toString();
      });
      
      process.on("close", (code) => {
        if (code === 0) {
          resolve(output);
        } else {
          reject(new Error(`nmap exited with code ${code}`));
        }
      });
    });
  }
  
  parseOutput(output: string): ParsedResult {
    // 解析XML输出
    const result: ParsedResult = {
      hosts: [],
      services: [],
    };
    
    // XML解析逻辑...
    
    return result;
  }
  
  async isAvailable(): Promise<boolean> {
    try {
      await this.execute(["--version"]);
      return true;
    } catch {
      return false;
    }
  }
}
```

## 注意事项

1. **权限**: 端口扫描可能需要root权限
2. **速率**: 扫描速率需要控制，避免触发IDS
3. **超时**: 扫描可能需要较长时间，设置合理超时
4. **结果解析**: 不同扫描器输出格式不同，需要分别解析
5. **CVE数据库**: 需要定期更新CVE数据库
