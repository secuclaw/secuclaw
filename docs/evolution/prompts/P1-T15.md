## 示例代码片段

### azure.ts 骨架

```typescript
import { OpenAIClient, AzureKeyCredential } from "@azure/openai";
import { BaseProvider, type Message, type ProviderConfig } from "./base.js";
import { AzureAuthManager, type AzureAuthMethod } from "./azure-auth.js";

export class AzureOpenAIProvider extends BaseProvider {
  private client: OpenAIClient;
  private authManager: AzureAuthManager;
  
  constructor(private config: AzureOpenAIConfig) {
    super(config);
    
    // 初始化认证
    const authMethod = this.resolveAuthMethod();
    this.authManager = new AzureAuthManager(authMethod);
    
    // 初始化客户端
    this.client = new OpenAIClient(config.endpoint, new AzureKeyCredential("placeholder"));
  }
  
  private resolveAuthMethod(): AzureAuthMethod {
    if (this.config.apiKey) {
      return { type: "api-key", key: this.config.apiKey };
    }
    if (this.config.useAzureAD) {
      return {
        type: "azure-ad",
        tenantId: this.config.tenantId!,
        clientId: this.config.clientId!,
        clientSecret: this.config.clientSecret,
      };
    }
    return { type: "managed-identity" };
  }
  
  async chat(messages: Message[]): Promise<string> {
    const token = await this.authManager.getAccessToken();
    
    const response = await this.client.getChatCompletions(
      this.config.deploymentName,
      messages.map(m => ({ role: m.role, content: m.content })),
      {
        apiVersion: this.config.apiVersion,
        accessToken: token,
        temperature: this.config.temperature,
        maxTokens: this.config.maxTokens,
      }
    );
    
    return response.choices[0].message?.content ?? "";
  }
  
  async *chatStream(messages: Message[]): AsyncIterable<string> {
    const token = await this.authManager.getAccessToken();
    
    const events = await this.client.streamChatCompletions(
      this.config.deploymentName,
      messages.map(m => ({ role: m.role, content: m.content })),
      {
        apiVersion: this.config.apiVersion,
        accessToken: token,
      }
    );
    
    for await (const event of events) {
      const delta = event.choices[0]?.delta?.content;
      if (delta) {
        yield delta;
      }
    }
  }
  
  getDeploymentInfo(): DeploymentInfo {
    return {
      endpoint: this.config.endpoint,
      deployment: this.config.deploymentName,
      apiVersion: this.config.apiVersion,
    };
  }
}
```

### azure-auth.ts 骨架

```typescript
import { DefaultAzureCredential, ClientSecretCredential } from "@azure/identity";

export type AzureAuthMethod = 
  | { type: "api-key"; key: string }
  | { type: "azure-ad"; tenantId: string; clientId: string; clientSecret?: string }
  | { type: "managed-identity" };

export class AzureAuthManager {
  private credential: DefaultAzureCredential | ClientSecretCredential | null = null;
  
  constructor(private method: AzureAuthMethod) {
    if (method.type === "azure-ad") {
      if (method.clientSecret) {
        this.credential = new ClientSecretCredential(
          method.tenantId,
          method.clientId,
          method.clientSecret
        );
      }
    }
  }
  
  async getAccessToken(): Promise<string> {
    if (this.method.type === "api-key") {
      return this.method.key;
    }
    
    if (!this.credential) {
      this.credential = new DefaultAzureCredential();
    }
    
    const token = await this.credential.getToken("https://cognitiveservices.azure.com/.default");
    return token.token;
  }
  
  async getAuthorizationHeader(): Promise<string> {
    const token = await this.getAccessToken();
    if (this.method.type === "api-key") {
      return `Bearer ${token}`;
    }
    return `Bearer ${token}`;
  }
}
```

## 注意事项

1. **认证优先级**: API密钥 > Azure AD > 托管标识
2. **Token缓存**: Azure AD token需要缓存，避免频繁请求
3. **部署名称**: Azure使用部署名称而非模型名称
4. **API版本**: 不同API版本功能可能不同
5. **区域限制**: 某些模型在特定区域不可用
