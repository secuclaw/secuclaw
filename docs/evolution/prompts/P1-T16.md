## 示例代码片段

### bedrock.ts 骨架

```typescript
import {
  BedrockRuntimeClient,
  InvokeModelCommand,
  InvokeModelWithResponseStreamCommand,
} from "@aws-sdk/client-bedrock-runtime";
import { BaseProvider, type Message, type ProviderConfig } from "./base.js";
import { BedrockAuthManager, type AWSAuthMethod } from "./bedrock-auth.js";
import { getBedrockModelConfig } from "./bedrock-models.js";

export class BedrockProvider extends BaseProvider {
  private client: BedrockRuntimeClient;
  private authManager: BedrockAuthManager;
  
  constructor(private config: BedrockConfig) {
    super(config);
    this.authManager = new BedrockAuthManager(this.resolveAuthMethod(), config.region);
    this.client = this.authManager.getClient();
  }
  
  private resolveAuthMethod(): AWSAuthMethod {
    if (config.accessKeyId && config.secretAccessKey) {
      return {
        type: "access-key",
        accessKeyId: config.accessKeyId,
        secretAccessKey: config.secretAccessKey,
        sessionToken: config.sessionToken,
      };
    }
    if (config.roleArn) {
      return { type: "iam-role", roleArn: config.roleArn };
    }
    return { type: "environment" };
  }
  
  async chat(messages: Message[]): Promise<string> {
    const modelConfig = getBedrockModelConfig(this.config.modelId);
    const provider = modelConfig?.provider ?? "anthropic";
    
    const body = this.buildRequestBody(messages, provider);
    
    const command = new InvokeModelCommand({
      modelId: this.config.modelId,
      contentType: "application/json",
      accept: "application/json",
      body: JSON.stringify(body),
    });
    
    const response = await this.client.send(command);
    const result = JSON.parse(new TextDecoder().decode(response.body));
    
    return this.extractContent(result, provider);
  }
  
  async *chatStream(messages: Message[]): AsyncIterable<string> {
    const modelConfig = getBedrockModelConfig(this.config.modelId);
    const provider = modelConfig?.provider ?? "anthropic";
    
    const body = this.buildRequestBody(messages, provider);
    
    const command = new InvokeModelWithResponseStreamCommand({
      modelId: this.config.modelId,
      contentType: "application/json",
      accept: "application/json",
      body: JSON.stringify(body),
    });
    
    const response = await this.client.send(command);
    
    for await (const event of response.body ?? []) {
      if (event.chunk?.bytes) {
        const chunk = JSON.parse(new TextDecoder().decode(event.chunk.bytes));
        const delta = this.extractStreamDelta(chunk, provider);
        if (delta) {
          yield delta;
        }
      }
    }
  }
  
  private buildRequestBody(messages: Message[], provider: string): unknown {
    if (provider === "anthropic") {
      return {
        anthropic_version: "bedrock-2023-05-31",
        messages: messages.map(m => ({ role: m.role, content: m.content })),
        max_tokens: this.config.maxTokens ?? 4096,
        temperature: this.config.temperature ?? 0.7,
      };
    }
    // 其他provider的请求格式...
    return { messages, max_gen_len: this.config.maxTokens };
  }
}
```

### bedrock-auth.ts 骨架

```typescript
import {
  BedrockRuntimeClient,
  BedrockClient,
  ListFoundationModelsCommand,
} from "@aws-sdk/client-bedrock";
import { fromEnv, fromTemporaryCredentials } from "@aws-sdk/credential-providers";

export type AWSAuthMethod = 
  | { type: "access-key"; accessKeyId: string; secretAccessKey: string; sessionToken?: string }
  | { type: "iam-role"; roleArn: string }
  | { type: "environment" }
  | { type: "ec2-metadata" };

export class BedrockAuthManager {
  private client: BedrockRuntimeClient | null = null;
  
  constructor(private method: AWSAuthMethod, private region: string) {}
  
  async getCredentials(): Promise<AwsCredentials> {
    switch (this.method.type) {
      case "access-key":
        return {
          accessKeyId: this.method.accessKeyId,
          secretAccessKey: this.method.secretAccessKey,
          sessionToken: this.method.sessionToken,
        };
      case "iam-role":
        const tempCreds = fromTemporaryCredentials({
          masterCredentials: fromEnv(),
          params: { RoleArn: this.method.roleArn, RoleSessionName: "secuclaw" },
        });
        return tempCreds();
      default:
        return fromEnv()();
    }
  }
  
  getClient(): BedrockRuntimeClient {
    if (!this.client) {
      this.client = new BedrockRuntimeClient({
        region: this.region,
        credentials: async () => this.getCredentials(),
      });
    }
    return this.client;
  }
}
```

## 注意事项

1. **区域限制**: Bedrock在部分区域不可用，需要检查区域支持
2. **模型可用性**: 不同模型在不同区域的可用性不同
3. **IAM权限**: 需要bedrock:InvokeModel权限
4. **限流**: Bedrock有TPM/RPM限制，需要处理限流错误
5. **Guardrails**: 企业环境可能需要配置Guardrails
