## 示例代码片段

### wake.ts 骨架

```typescript
import { PorcupineAdapter } from './porcupine-adapter';
import type { WakeCallback, WakeEvent, WakeConfig } from './types';

export class VoiceWakeDetector {
  private adapter: PorcupineAdapter;
  private callbacks: Set<WakeCallback> = new Set();
  private isRunning = false;
  
  constructor(config: WakeConfig) {
    this.adapter = new PorcupineAdapter(config.accessKey, config.wakeWords);
  }

  async start(): Promise<void> {
    if (this.isRunning) return;
    this.isRunning = true;
    await this.adapter.init();
    this.startListening();
  }

  async stop(): Promise<void> {
    this.isRunning = false;
    await this.adapter.release();
  }

  onWake(callback: WakeCallback): void {
    this.callbacks.add(callback);
  }

  async setWakeWord(word: string): Promise<void> {
    await this.adapter.addWakeWord(word);
  }

  private startListening(): void {
    // 获取麦克风输入并处理
  }

  private handleWakeEvent(event: WakeEvent): void {
    for (const callback of this.callbacks) {
      callback(event);
    }
  }
}
```

### porcupine-adapter.ts 骨架

```typescript
import { Porcupine } from '@picovoice/porcupine-node';

export class PorcupineAdapter {
  private porcupine: Porcupine | null = null;
  private accessKey: string;
  private wakeWords: string[];
  
  constructor(accessKey: string, wakeWords: string[]) {
    this.accessKey = accessKey;
    this.wakeWords = wakeWords;
  }

  async init(): Promise<void> {
    this.porcupine = new Porcupine(
      this.accessKey,
      this.wakeWords.map(word => ({ builtin: word }))
    );
  }

  process(audioFrame: Int16Array): number {
    if (!this.porcupine) return -1;
    return this.porcupine.process(audioFrame);
  }

  async release(): Promise<void> {
    if (this.porcupine) {
      this.porcupine.release();
      this.porcupine = null;
    }
  }
}
```
