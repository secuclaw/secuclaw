## 示例代码片段

### bluebubbles-client.ts 骨架

```typescript
import { BaseChannel, ChannelConfig, ChannelStatus } from '../base';
import type { IMessageConfig, IMessageMessage } from './types';

export class BlueBubblesClient extends BaseChannel {
  private serverUrl: string;
  private apiToken: string;
  
  constructor(config: IMessageConfig) {
    super(config);
    this.serverUrl = config.serverUrl;
    this.apiToken = config.apiToken;
  }

  async sendMessage(chatId: string, message: string): Promise<string> {
    const response = await fetch(`${this.serverUrl}/api/v1/message/text`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.apiToken}`,
      },
      body: JSON.stringify({
        chatGuid: chatId,
        message: message,
      }),
    });
    const data = await response.json();
    return data.message.guid;
  }

  async getMessages(chatId: string, limit: number = 50): Promise<IMessageMessage[]> {
    const response = await fetch(
      `${this.serverUrl}/api/v1/chat/${chatId}/message?limit=${limit}`,
      {
        headers: { 'Authorization': `Bearer ${this.apiToken}` },
      }
    );
    return response.json();
  }

  async getChats(): Promise<BlueBubblesChat[]> {
    const response = await fetch(`${this.serverUrl}/api/v1/chat`, {
      headers: { 'Authorization': `Bearer ${this.apiToken}` },
    });
    return response.json();
  }
}
```
