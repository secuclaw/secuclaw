## 示例代码片段

### index.ts 骨架

```typescript
import { BaseHand } from "../base.js";
import type { HandContext, HandResult } from "../types.js";
import { ThreatHunterDefinition } from "./definition.js";
import { MitreMapper } from "./mitre-mapper.js";
import { IOCDetector } from "./ioc-detector.js";
import { LogAnalyzer } from "./log-analyzer.js";
import { CorrelationEngine } from "./correlation.js";
import { ReportGenerator } from "./report-generator.js";

export class ThreatHunterHand extends BaseHand {
  private mitreMapper?: MitreMapper;
  private iocDetector?: IOCDetector;
  private logAnalyzer?: LogAnalyzer;
  private correlationEngine?: CorrelationEngine;
  private reportGenerator?: ReportGenerator;
  
  constructor() {
    super(ThreatHunterDefinition);
  }
  
  static getDefinition() {
    return ThreatHunterDefinition;
  }
  
  async initialize(): Promise<void> {
    this.reportProgress(0, "初始化威胁狩猎引擎...");
    
    // 加载MITRE数据
    this.mitreMapper = new MitreMapper(await this.loadMitreData());
    this.reportProgress(20, "MITRE ATT&CK数据加载完成");
    
    // 初始化组件
    this.iocDetector = new IOCDetector();
    this.logAnalyzer = new LogAnalyzer();
    this.correlationEngine = new CorrelationEngine();
    this.reportGenerator = new ReportGenerator();
    this.reportProgress(40, "分析组件初始化完成");
  }
  
  async execute(context: HandContext): Promise<HandResult> {
    const startTime = Date.now();
    
    try {
      // 1. 采集日志数据
      this.reportProgress(50, "采集安全日志...");
      const logs = await this.collectLogs(context);
      
      // 2. IOC检测
      this.reportProgress(60, "执行IOC检测...");
      const iocs = await this.iocDetector!.scanLogs(logs);
      
      // 3. 事件关联
      this.reportProgress(70, "关联安全事件...");
      const events = this.logAnalyzer!.extractEvents(logs);
      const correlatedGroups = this.correlationEngine!.correlateByTime(
        events,
        context.config.correlationWindow * 3600000
      );
      
      // 4. MITRE映射
      this.reportProgress(80, "映射MITRE ATT&CK...");
      const techniques = await this.mapToMitre(iocs, correlatedGroups);
      
      // 5. 风险评分
      this.reportProgress(90, "计算风险评分...");
      const riskScores = correlatedGroups.map(g => 
        this.correlationEngine!.calculateRiskScore(g)
      );
      
      // 6. 生成报告
      const result: ThreatHuntResult = {
        iocs,
        correlatedGroups,
        techniques,
        riskScores,
        summary: this.generateSummary(iocs, correlatedGroups, techniques),
      };
      
      const report = this.reportGenerator!.generateMarkdown(result);
      
      this.reportProgress(100, "威胁狩猎完成");
      
      return {
        success: true,
        data: result,
        duration: Date.now() - startTime,
        metrics: {
          threats_detected: iocs.length,
          correlated_events: correlatedGroups.length,
          mitre_coverage: techniques.length / 14, // 14 tactics
        },
        artifacts: [
          { type: "report", name: "threat-report.md", content: report },
        ],
      };
    } catch (error) {
      return {
        success: false,
        error: { code: "EXECUTION_ERROR", message: String(error), recoverable: true },
        duration: Date.now() - startTime,
        metrics: {},
      };
    }
  }
  
  async terminate(): Promise<void> {
    this.mitreMapper = undefined;
    this.iocDetector = undefined;
    this.logAnalyzer = undefined;
    this.correlationEngine = undefined;
    this.reportGenerator = undefined;
  }
}
```

## 注意事项

1. **性能**: 大量日志分析需要分批处理
2. **准确性**: IOC检测需要验证，避免误报
3. **隐私**: 日志中可能包含敏感信息，需要脱敏
4. **关联窗口**: 合理设置关联时间窗口
5. **报告格式**: 支持多种输出格式（Markdown、JSON、HTML）
