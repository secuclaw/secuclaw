## 示例代码片段

### cli-wrapper.ts 骨架

```typescript
import { spawn } from 'child_process';
import { BaseChannel, ChannelConfig, ChannelStatus } from '../base';
import type { SignalConfig } from './types';

export class SignalCLI extends BaseChannel {
  private signalCliPath: string;
  private phoneNumber: string;
  
  constructor(config: SignalConfig) {
    super(config);
    this.signalCliPath = config.signalCliPath || 'signal-cli';
    this.phoneNumber = config.phoneNumber;
  }

  async send(recipient: string, message: string): Promise<void> {
    await this.execute([
      '-u', this.phoneNumber,
      'send', recipient,
      '-m', message
    ]);
  }

  async receive(): Promise<SignalMessage[]> {
    const output = await this.execute([
      '-u', this.phoneNumber,
      'receive', '--json'
    ]);
    return JSON.parse(output);
  }

  private async execute(args: string[]): Promise<string> {
    return new Promise((resolve, reject) => {
      const proc = spawn(this.signalCliPath, args);
      let stdout = '';
      let stderr = '';
      
      proc.stdout.on('data', (data) => stdout += data);
      proc.stderr.on('data', (data) => stderr += data);
      
      proc.on('close', (code) => {
        if (code === 0) resolve(stdout);
        else reject(new Error(stderr));
      });
    });
  }
}
```

## 注意事项

1. Signal使用signal-cli作为后端，需要通过命令行交互
2. 首次使用需要注册和验证电话号码
3. 端到端加密由Signal协议自动处理
4. 需要安装signal-cli到系统PATH或指定路径
