## 示例代码片段

### pool.ts 骨架

```typescript
export class ObjectPool<T> {
  private pool: T[] = [];
  private factory: () => T;
  private reset: (obj: T) => void;
  private maxSize: number;

  constructor(
    factory: () => T,
    reset: (obj: T) => void,
    maxSize: number = 100
  ) {
    this.factory = factory;
    this.reset = reset;
    this.maxSize = maxSize;
  }

  /**
   * 获取对象
   */
  acquire(): T {
    const obj = this.pool.pop();
    return obj !== undefined ? obj : this.factory();
  }

  /**
   * 释放对象
   */
  release(obj: T): void {
    if (this.pool.length < this.maxSize) {
      this.reset(obj);
      this.pool.push(obj);
    }
  }

  /**
   * 清空池
   */
  drain(): void {
    this.pool = [];
  }

  /**
   * 获取池大小
   */
  size(): number {
    return this.pool.length;
  }
}
```

### gc-hints.ts 骨架

```typescript
export class GCHints {
  private gcInterval?: NodeJS.Timeout;
  private frequency: number;

  constructor(frequency: number = 60000) {
    this.frequency = frequency;
  }

  /**
   * 触发GC
   */
  triggerGC(): void {
    if (global.gc) {
      global.gc();
    }
  }

  /**
   * 计划GC
   */
  scheduleGC(): void {
    this.gcInterval = setInterval(() => {
      this.triggerGC();
    }, this.frequency);
  }

  /**
   * 停止GC
   */
  stopGC(): void {
    if (this.gcInterval) {
      clearInterval(this.gcInterval);
    }
  }

  /**
   * 设置频率
   */
  setGCFrequency(frequency: number): void {
    this.frequency = frequency;
    this.stopGC();
    this.scheduleGC();
  }
}
```

### monitor.ts 骨架

```typescript
export class MemoryMonitor {
  private samples: Array<{ timestamp: number; usage: NodeJS.MemoryUsage }> = [];
  private interval?: NodeJS.Timeout;

  /**
   * 开始监控
   */
  startMonitoring(intervalMs: number = 1000): void {
    this.interval = setInterval(() => {
      this.samples.push({
        timestamp: Date.now(),
        usage: process.memoryUsage(),
      });

      // 保留最近1000个样本
      if (this.samples.length > 1000) {
        this.samples.shift();
      }
    }, intervalMs);
  }

  /**
   * 停止监控
   */
  stopMonitoring(): void {
    if (this.interval) {
      clearInterval(this.interval);
    }
  }

  /**
   * 获取统计
   */
  getStats(): MemoryStats {
    if (this.samples.length === 0) {
      return {
        avgHeapUsed: 0,
        maxHeapUsed: 0,
        minHeapUsed: 0,
        samples: 0,
      };
    }

    const heapUseds = this.samples.map(s => s.usage.heapUsed);
    
    return {
      avgHeapUsed: heapUseds.reduce((a, b) => a + b, 0) / heapUseds.length,
      maxHeapUsed: Math.max(...heapUseds),
      minHeapUsed: Math.min(...heapUseds),
      samples: this.samples.length,
    };
  }

  /**
   * 检测泄漏
   */
  detectLeak(): boolean {
    if (this.samples.length < 10) return false;

    // 检查内存是否持续增长
    const recent = this.samples.slice(-10);
    const trend = recent.map(s => s.usage.heapUsed);
    
    let increasing = 0;
    for (let i = 1; i < trend.length; i++) {
      if (trend[i] > trend[i - 1]) increasing++;
    }

    return increasing > 7; // 70%以上在增长
  }
}
```

## 注意事项

1. **对象池**: 复用对象，减少分配
2. **GC提示**: 定期触发GC，避免内存峰值
3. **泄漏检测**: 及时发现泄漏
4. **监控**: 实时监控内存使用
5. **基准测试**: 建立性能基准
