## 示例代码片段

### base.ts 骨架

```typescript
export interface TTSOptions {
  voice?: string;
  speed?: number;
  pitch?: number;
  language?: string;
}

export interface Voice {
  id: string;
  name: string;
  language: string;
  gender?: 'male' | 'female' | 'neutral';
}

export abstract class TTSProvider {
  abstract getName(): string;
  abstract synthesize(text: string, options?: TTSOptions): Promise<Buffer>;
  abstract synthesizeStream(text: string, options?: TTSOptions): AsyncIterable<Buffer>;
  abstract getVoices(): Promise<Voice[]>;
}
```

### factory.ts 骨架

```typescript
import { TTSProvider } from './base';
import { ElevenLabsProvider } from './elevenlabs';
import { OpenAITTSProvider } from './openai';
import { EdgeTTSProvider } from './edge';

export class TTSFactory {
  private static providers = new Map<string, new (config: any) => TTSProvider>();
  
  static register(name: string, provider: new (config: any) => TTSProvider): void {
    this.providers.set(name, provider);
  }
  
  static create(name: string, config: any): TTSProvider {
    const Provider = this.providers.get(name);
    if (!Provider) throw new Error(`Unknown TTS provider: ${name}`);
    return new Provider(config);
  }
  
  static getAvailableProviders(): string[] {
    return Array.from(this.providers.keys());
  }
}

// 注册默认提供商
TTSFactory.register('elevenlabs', ElevenLabsProvider);
TTSFactory.register('openai', OpenAITTSProvider);
TTSFactory.register('edge', EdgeTTSProvider);
```
