## 示例代码片段

### loader.ts 骨架

```typescript
import { ModuleCache } from './cache';
import { ModuleResolver } from './resolver';

export class LazyLoader {
  private cache: ModuleCache;
  private resolver: ModuleResolver;
  private loading: Map<string, Promise<any>>;

  constructor() {
    this.cache = new ModuleCache(100);
    this.resolver = new ModuleResolver();
    this.loading = new Map();
  }

  /**
   * 加载模块
   */
  async load(moduleName: string): Promise<any> {
    // 检查缓存
    if (this.cache.has(moduleName)) {
      return this.cache.get(moduleName);
    }

    // 检查是否正在加载
    if (this.loading.has(moduleName)) {
      return this.loading.get(moduleName);
    }

    // 解析模块路径
    const modulePath = this.resolver.resolve(moduleName);
    
    // 开始加载
    const loadPromise = this.loadModule(modulePath);
    this.loading.set(moduleName, loadPromise);

    try {
      const module = await loadPromise;
      this.cache.set(moduleName, module);
      return module;
    } finally {
      this.loading.delete(moduleName);
    }
  }

  /**
   * 预加载模块
   */
  async preload(modules: string[]): Promise<void> {
    await Promise.all(modules.map(m => this.load(m)));
  }

  /**
   * 卸载模块
   */
  unload(moduleName: string): void {
    this.cache.delete(moduleName);
  }

  /**
   * 检查是否已加载
   */
  isLoaded(moduleName: string): boolean {
    return this.cache.has(moduleName);
  }

  private async loadModule(path: string): Promise<any> {
    return await import(path);
  }
}
```

## 懒加载示例

```typescript
class LazyLoader {
  private cache = new Map<string, any>();
  
  async load(moduleName: string): Promise<any> {
    if (this.cache.has(moduleName)) {
      return this.cache.get(moduleName);
    }
    
    const module = await import(`./modules/${moduleName}`);
    this.cache.set(moduleName, module);
    return module;
  }
}
```
