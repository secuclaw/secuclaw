- 不要混合v0.8和v0.9消息
- 不要阻塞主线程

## 示例代码片段

### protocol.ts 骨架

```typescript
const A2UI_ACTION_KEYS = [
  "beginRendering",
  "surfaceUpdate",
  "dataModelUpdate",
  "deleteSurface",
  "createSurface",
] as const;

export type A2UIVersion = "v0.8" | "v0.9";
export type A2UIAction = typeof A2UI_ACTION_KEYS[number];

export interface A2UIMessage {
  version: A2UIVersion;
  action: A2UIAction;
  payload: Record<string, unknown>;
}

export class A2UIProtocol {
  constructor(public version: A2UIVersion = "v0.8") {}
  
  encode(message: A2UIMessage): string {
    const payload = { [message.action]: message.payload };
    return JSON.stringify(payload);
  }
  
  decode(data: string): A2UIMessage {
    const obj = JSON.parse(data) as Record<string, unknown>;
    const actionKeys = A2UI_ACTION_KEYS.filter((key) => key in obj);
    if (actionKeys.length !== 1) {
      throw new Error(`Expected exactly one action key, got ${actionKeys.length}`);
    }
    const action = actionKeys[0];
    const version = action === "createSurface" ? "v0.9" : "v0.8";
    return { version, action, payload: obj[action] as Record<string, unknown> };
  }
  
  encodeJsonl(messages: A2UIMessage[]): string {
    return messages.map((msg) => this.encode(msg)).join("\n");
  }
  
  decodeJsonl(jsonl: string): A2UIMessage[] {
    const lines = jsonl.split(/\r?\n/);
    const messages: A2UIMessage[] = [];
    for (const line of lines) {
      const trimmed = line.trim();
      if (trimmed) {
        messages.push(this.decode(trimmed));
      }
    }
    return messages;
  }
}
```

### client.ts 桥接实现

```typescript
export class A2UIClient {
  private handlers = new Set<MessageHandler>();
  
  constructor(private config: A2UIClientConfig) {}
  
  // 跨平台桥接
  private postToNode(payload: unknown): boolean {
    const raw = typeof payload === "string" ? payload : JSON.stringify(payload);
    
    // iOS WebView
    const iosHandler = (globalThis as any).webkit?.messageHandlers?.secuclawA2UIAction;
    if (iosHandler && typeof iosHandler.postMessage === "function") {
      iosHandler.postMessage(raw);
      return true;
    }
    
    // Android WebView
    const androidHandler = (globalThis as any).secuclawA2UIAction;
    if (androidHandler && typeof androidHandler.postMessage === "function") {
      androidHandler.postMessage(raw);
      return true;
    }
    
    // WebSocket fallback
    if (this.config.wsClient) {
      this.config.wsClient.send(raw);
      return true;
    }
    
    return false;
  }
  
  sendUserAction(action: UserAction): boolean {
    const id = action.id || (globalThis.crypto?.randomUUID?.() ?? String(Date.now()));
    return this.postToNode({ userAction: { ...action, id } });
  }
  
  onMessage(handler: MessageHandler): () => void {
    this.handlers.add(handler);
    return () => this.handlers.delete(handler);
  }
}
```

export function buildThreatDashboard(threats: Threat[]): A2UIMessage[] {
  const surfaceId = "main";
  const rootId = "root";
  const cardIds = threats.map((_, i) => `threat-${i}`);
  
  return [
    // Surface更新
    {
      version: "v0.8",
      action: "surfaceUpdate",
      payload: {

        components: [

            component: { ThreatCard: t },
          })),
        ],
      },
    },
    // 开始渲染
    {
      version: "v0.8",
      action: "beginRendering",
      payload: { surfaceId, root: rootId },
    },
  ];

## 注意事项

1. **版本兼容**: v0.8和v0.9消息不能混用，需要严格验证
2. **桥接差异**: iOS和Android WebView桥接API不同，需要分别处理
3. **消息ID**: 每个用户动作必须有唯一ID，用于状态回调
4. **错误恢复**: 协议解析失败时需要优雅降级
5. **性能优化**: 大量组件更新时考虑批量处理
