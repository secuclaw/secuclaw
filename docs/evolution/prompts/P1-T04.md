## 示例代码片段

### api-client.ts 骨架

```typescript
import { BaseChannel, ChannelConfig, ChannelStatus } from '../base';
import type { GoogleChatConfig, GoogleChatMessage } from './types';

export class GoogleChatChannel extends BaseChannel {
  private credentials: any;
  
  constructor(config: GoogleChatConfig) {
    super(config);
    this.credentials = config.credentials;
  }

  async sendMessage(spaceId: string, message: string | GoogleChatCard): Promise<string> {
    const response = await fetch(
      `https://chat.googleapis.com/v1/${spaceId}/messages`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${await this.getAccessToken()}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          text: typeof message === 'string' ? message : undefined,
          cardsV2: typeof message !== 'string' ? [{ cardId: 'card1', card: message }] : undefined,
        }),
      }
    );
    const data = await response.json();
    return data.name;
  }

  async listSpaces(): Promise<GoogleChatSpace[]> {
    const response = await fetch('https://chat.googleapis.com/v1/spaces', {
      headers: { 'Authorization': `Bearer ${await this.getAccessToken()}` },
    });
    const data = await response.json();
    return data.spaces || [];
  }

  private async getAccessToken(): Promise<string> {
    // OAuth 2.0 token获取逻辑
    return '';
  }
}
```
