## 示例代码片段

### auth.ts 骨架

```typescript
import { ChallengeManager } from './challenge';
import { SessionManager } from './session';
import { NonceManager } from './nonce';
import { ReplayDefense } from './replay-defense';
import { HandshakeRequest, HandshakeResponse, Session } from './types';

export class P2PAuthenticator {
  private challengeManager: ChallengeManager;
  private sessionManager: SessionManager;
  private nonceManager: NonceManager;
  private replayDefense: ReplayDefense;
  private privateKey: string;
  private peerPublicKeys: Map<string, string>;

  constructor(privateKey: string) {
    this.challengeManager = new ChallengeManager();
    this.sessionManager = new SessionManager();
    this.nonceManager = new NonceManager();
    this.replayDefense = new ReplayDefense();
    this.privateKey = privateKey;
    this.peerPublicKeys = new Map();
  }

  /**
   * 初始化握手（Agent A调用）
   */
  async initiateHandshake(peerId: string): Promise<HandshakeRequest> {
    // 生成Nonce_A
    const nonceA = await this.nonceManager.generateNonce();
    
    // 创建握手请求
    const request: HandshakeRequest = {
      version: '1.0',
      from: this.getMyId(),
      to: peerId,
      nonceA,
      timestamp: Date.now(),
    };
    
    // 签名请求
    const signature = await this.signRequest(request);
    request.signature = signature;
    
    return request;
  }

  /**
   * 处理握手请求（Agent B调用）
   */
  async handleHandshake(request: HandshakeRequest): Promise<HandshakeResponse> {
    // 验证请求签名
    const valid = await this.verifyRequestSignature(request);
    if (!valid) {
      throw new Error('Invalid request signature');
    }
    
    // 检查重放
    if (this.replayDefense.checkReplay(request)) {
      throw new Error('Replay attack detected');
    }
    
    // 验证Nonce
    if (!await this.nonceManager.validateNonce(request.nonceA)) {
      throw new Error('Invalid nonce');
    }
    
    // 生成Nonce_B和Challenge_A
    const nonceB = await this.nonceManager.generateNonce();
    const challengeA = await this.challengeManager.generateChallenge(request.nonceA);
    
    // 创建响应
    const response: HandshakeResponse = {
      version: '1.0',
      from: this.getMyId(),
      to: request.from,
      nonceB,
      challengeA,
      responseA: await this.createChallengeResponse(challengeA, request.nonceA),
      timestamp: Date.now(),
    };
    
    // 签名响应
    const signature = await this.signResponse(response);
    response.signature = signature;
    
    // 记录消息（防重放）
    this.replayDefense.recordMessage(request);
    
    return response;
  }

  /**
   * 完成握手（Agent A调用）
   */
  async completeHandshake(response: HandshakeResponse): Promise<Session> {
    // 验证响应签名
    const valid = await this.verifyResponseSignature(response);
    if (!valid) {
      throw new Error('Invalid response signature');
    }
    
    // 检查重放
    if (this.replayDefense.checkReplay(response)) {
      throw new Error('Replay attack detected');
    }
    
    // 验证Challenge_A响应
    const challengeValid = await this.challengeManager.verifyResponse(
      response.challengeA,
      response.responseA,
      this.privateKey
    );
    if (!challengeValid) {
      throw new Error('Invalid challenge response');
    }
    
    // 生成Challenge_B并响应
    const challengeB = await this.challengeManager.generateChallenge(response.nonceB);
    const responseB = await this.createChallengeResponse(challengeB, response.nonceB);
    
    // 创建会话
    const session = await this.sessionManager.createSession({
      peerId: response.from,
      nonceA: response.nonceB, // 使用对方的nonce
      nonceB: response.nonceB,
      challengeB,
      responseB,
    });
    
    // 记录消息（防重放）
    this.replayDefense.recordMessage(response);
    
    return session;
  }

  /**
   * 验证会话
   */
  async verifySession(sessionId: string): Promise<boolean> {
    const session = await this.sessionManager.getSession(sessionId);
    if (!session) {
      return false;
    }
    
    // 检查会话是否过期
    if (session.expiresAt < Date.now()) {
      await this.sessionManager.terminateSession(sessionId);
      return false;
    }
    
    return true;
  }

  private getMyId(): string {
    // 获取自己的ID
    return 'my-agent-id';
  }

  private async signRequest(request: HandshakeRequest): Promise<string> {
    // 使用私钥签名请求
    return 'signature';
  }

  private async verifyRequestSignature(request: HandshakeRequest): Promise<boolean> {
    // 验证请求签名
    return true;
  }

  private async signResponse(response: HandshakeResponse): Promise<string> {
    // 使用私钥签名响应
    return 'signature';
  }

  private async verifyResponseSignature(response: HandshakeResponse): Promise<boolean> {
    // 验证响应签名
    return true;
  }

  private async createChallengeResponse(challenge: string, nonce: string): Promise<string> {
    // 创建挑战响应
    return await this.challengeManager.createResponse(challenge, nonce, this.privateKey);
  }
}
```

### challenge.ts 骨架

```typescript
import crypto from 'crypto';
import { HMACVerifier } from './hmac';

export class ChallengeManager {
  private hmacVerifier: HMACVerifier;

  constructor() {
    this.hmacVerifier = new HMACVerifier();
  }

  /**
   * 生成挑战
   */
  async generateChallenge(nonce: string): Promise<string> {
    // 生成随机挑战
    const randomBytes = crypto.randomBytes(32);
    const challenge = randomBytes.toString('hex');
    
    // 结合nonce创建最终挑战
    const combined = `${nonce}:${challenge}`;
    return combined;
  }

  /**
   * 创建挑战响应
   */
  async createResponse(
    challenge: string,
    nonce: string,
    secretKey: string
  ): Promise<string> {
    // 使用HMAC-SHA256创建响应
    const data = `${challenge}:${nonce}`;
    const hmac = await this.hmacVerifier.computeHMAC(data, secretKey);
    return hmac;
  }

  /**
   * 验证挑战响应
   */
  async verifyResponse(
    challenge: string,
    response: string,
    secretKey: string
  ): Promise<boolean> {
    // 从挑战中提取nonce
    const parts = challenge.split(':');
    const nonce = parts[0];
    
    // 重新计算HMAC
    const data = `${challenge}:${nonce}`;
    const expectedHMAC = await this.hmacVerifier.computeHMAC(data, secretKey);
    
    // 常量时间比较
    return this.hmacVerifier.constantTimeCompare(response, expectedHMAC);
  }
}
```

### hmac.ts 骨架

```typescript
import crypto from 'crypto';

export class HMACVerifier {
  /**
   * 计算HMAC-SHA256
   */
  async computeHMAC(data: string, key: string): Promise<string> {
    const hmac = crypto.createHmac('sha256', key);
    hmac.update(data);
    return hmac.digest('hex');
  }

  /**
   * 验证HMAC
   */
  async verifyHMAC(data: string, key: string, expectedHMAC: string): Promise<boolean> {
    const computedHMAC = await this.computeHMAC(data, key);
    return this.constantTimeCompare(computedHMAC, expectedHMAC);
  }

  /**
   * 常量时间比较
   */
  constantTimeCompare(a: string, b: string): boolean {
    if (a.length !== b.length) {
      return false;
    }
    
    const bufferA = Buffer.from(a, 'hex');
    const bufferB = Buffer.from(b, 'hex');
    
    let result = 0;
    for (let i = 0; i < bufferA.length; i++) {
      result |= bufferA[i] ^ bufferB[i];
    }
    
    return result === 0;
  }
}
```

## 认证流程

```
┌─────────┐                              ┌─────────┐
│ Agent A │                              │ Agent B │
└────┬────┘                              └────┬────┘
     │                                        │
     │  1. Hello + Nonce_A                    │
     │───────────────────────────────────────►│
     │                                        │
     │  2. Hello + Nonce_B + Challenge_A      │
     │◄───────────────────────────────────────│
     │                                        │
     │  3. Response_A + Challenge_B           │
     │───────────────────────────────────────►│
     │                                        │
     │  4. Response_B                         │
     │◄───────────────────────────────────────│
     │                                        │
     │  5. Session Established                │
     │◄──────────────────────────────────────►│
```

## 注意事项

1. **双向认证**: 双方都必须验证对方身份
2. **Nonce唯一**: 每个Nonce只能使用一次
3. **常量时间**: 所有比较使用常量时间算法
4. **防重放**: 记录并检查所有消息
5. **会话管理**: 会话必须有合理的过期时间
6. **密钥安全**: 共享密钥必须安全存储
