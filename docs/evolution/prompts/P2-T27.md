### key-manager.ts 骨架

```typescript
import * as ed from '@noble/ed25519';
import { KeyPair, PrivateKey, PublicKey } from './types';

export class KeyManager {
  private keys: Map<string, KeyPair>;

  constructor() {
    this.keys = new Map();
  }

  /**
   * 生成Ed25519密钥对
   */
  async generateKeyPair(): Promise<KeyPair> {
    // 生成私钥（32字节）
    const privateKey = ed.utils.randomPrivateKey();
    
    // 从私钥派生公钥
    const publicKey = await ed.getPublicKeyAsync(privateKey);
    
    return {
      privateKey: this.bufferToHex(privateKey),
      publicKey: this.bufferToHex(publicKey),
      createdAt: new Date(),
    };
  }

  /**
   * 存储密钥对
   */
  async storeKeyPair(id: string, keyPair: KeyPair): Promise<void> {
    // 加密私钥后存储
    const encryptedPrivateKey = await this.encryptPrivateKey(keyPair.privateKey);
    
    this.keys.set(id, {
      ...keyPair,
      privateKey: encryptedPrivateKey,
    });
  }

  /**
   * 加载密钥对
   */
  async loadKeyPair(id: string): Promise<KeyPair> {
    const keyPair = this.keys.get(id);
    if (!keyPair) {
      throw new Error(`Key pair not found: ${id}`);
    }
    
    // 解密私钥
    const decryptedPrivateKey = await this.decryptPrivateKey(keyPair.privateKey);
    
    return {
      ...keyPair,
      privateKey: decryptedPrivateKey,
    };
  }

  /**
   * 删除密钥对
   */
  async deleteKeyPair(id: string): Promise<void> {
    const keyPair = this.keys.get(id);
    if (keyPair) {
      // 安全删除私钥（覆盖内存）
      this.secureDelete(keyPair.privateKey);
      this.keys.delete(id);
    }
  }

  /**
   * 导出公钥
   */
  exportPublicKey(id: string): string {
    const keyPair = this.keys.get(id);
    if (!keyPair) {
      throw new Error(`Key pair not found: ${id}`);
    }
    return keyPair.publicKey;
  }

  /**
   * 加密私钥
   */
  private async encryptPrivateKey(privateKey: string): Promise<string> {

   * 解密私钥
   */

    // 使用AES-256-GCM解密
    return encrypted; // 简化示例
  }

  /**
   * 安全删除（覆盖内存）
   */

    // 覆盖内存中的密钥
    // 实际实现需要更复杂的内存管理
```

### signer.ts 骨架

```typescript
import * as ed from '@noble/ed25519';
import { Signer as ISigner, Signature, Manifest } from './types';

export class Signer implements ISigner {
  private privateKey: Uint8Array;

  constructor(privateKey: string) {
    this.privateKey = this.hexToBuffer(privateKey);
  }

  /**
   * 签名数据
   */
  async sign(data: string | Uint8Array): Promise<Signature> {
    const message = typeof data === 'string' 
      ? new TextEncoder().encode(data)
      : data;
    
    // Ed25519签名
    const signature = await ed.signAsync(message, this.privateKey);
    
    return {
      value: this.bufferToHex(signature),
      algorithm: 'Ed25519',
      timestamp: new Date(),
    };
  }

  /**
   * 签名清单
   */
  async signManifest(manifest: Manifest): Promise<Manifest> {
    // 序列化清单（不包含签名）
    const manifestData = this.serializeManifest(manifest);
    
    // 签名
    const signature = await this.sign(manifestData);
    
    return {
      ...manifest,
      signature,
    };
  }

  /**
   * 签名消息
   */
  async signMessage(message: string): Promise<string> {
    const signature = await this.sign(message);
    return signature.value;
  }

  /**
   * 序列化清单
   */
  private serializeManifest(manifest: Manifest): string {
    const { signature, ...data } = manifest;
    return JSON.stringify(data, Object.keys(data).sort());
  }

  /**
   * Hex转Buffer
   */
  private hexToBuffer(hex: string): Uint8Array {
    const bytes = new Uint8Array(hex.length / 2);
    for (let i = 0; i < bytes.length; i++) {
      bytes[i] = parseInt(hex.substr(i * 2, 2), 16);
    }
    return bytes;
  }

  /**
   * Buffer转Hex
   */
  private bufferToHex(buffer: Uint8Array): string {
    return Array.from(buffer)
      .map(b => b.toString(16).padStart(2, '0'))
      .join('');
  }
}
```

### verifier.ts 骨架

```typescript
import * as ed from '@noble/ed25519';
import { Verifier as IVerifier, Signature, Manifest } from './types';

export class Verifier implements IVerifier {
  /**
   * 验证签名
   */
  async verify(
    data: string | Uint8Array,
    signature: Signature,
    publicKey: string
  ): Promise<boolean> {
    const message = typeof data === 'string'
      ? new TextEncoder().encode(data)
      : data;
    
    const sig = this.hexToBuffer(signature.value);
    const pub = this.hexToBuffer(publicKey);
    
    // Ed25519验证
    return await ed.verifyAsync(sig, message, pub);
  }

  /**
   * 验证清单
   */
  async verifyManifest(manifest: Manifest): Promise<boolean> {
    if (!manifest.signature) {
      return false;
    }
    
    // 序列化清单（不包含签名）
    const manifestData = this.serializeManifest(manifest);
    
    // 验证签名
    return await this.verify(manifestData, manifest.signature, manifest.publicKey);
  }

  /**
   * 验证消息
   */
  async verifyMessage(
    message: string,
    signature: string,
    publicKey: string
  ): Promise<boolean> {
    return await this.verify(message, {
      value: signature,
      algorithm: 'Ed25519',
      timestamp: new Date(),
    }, publicKey);
  }

  /**
   * 常量时间比较
   */
  constantTimeCompare(a: string, b: string): boolean {
    const bufferA = this.hexToBuffer(a);
    const bufferB = this.hexToBuffer(b);
    
    if (bufferA.length !== bufferB.length) {
      return false;
    }
    
    let result = 0;
    for (let i = 0; i < bufferA.length; i++) {
      result |= bufferA[i] ^ bufferB[i];
    }
    
    return result === 0;
  }

  /**
   * 序列化清单
   */
  private serializeManifest(manifest: Manifest): string {
    const { signature, ...data } = manifest;
    return JSON.stringify(data, Object.keys(data).sort());
  }

  /**
   * Hex转Buffer
   */
  private hexToBuffer(hex: string): Uint8Array {
    const bytes = new Uint8Array(hex.length / 2);
    for (let i = 0; i < bytes.length; i++) {
      bytes[i] = parseInt(hex.substr(i * 2, 2), 16);
    }
    return bytes;
  }
}
```

## 注意事项

1. **私钥安全**: 私钥必须加密存储，使用后立即清除内存
2. **常量时间**: 所有比较操作使用常量时间算法，防止时序攻击
3. **密钥轮换**: 支持平滑过渡，避免服务中断
4. **撤销机制**: 使用CRL或OCSP检查身份有效性
5. **签名算法**: 使用Ed25519，速度快且安全
6. **过期时间**: 清单必须设置合理的过期时间
