```typescript
import { BaseHand, HandContext, HandResult, HandDefinition } from '../base';
import { MISPAdapter } from './misp-adapter';
import { OTXAdapter } from './otx-adapter';
import { TAXIIAdapter } from './taxii-adapter';
import { IntelAggregator } from './intel-aggregator';
import { IOCExtractor } from './ioc-extractor';
import type { IntelSource, IntelData, IOC, AggregatedIntel } from './types';

export class ThreatIntelCollectorHand extends BaseHand {
  private mispAdapter?: MISPAdapter;
  private otxAdapter?: OTXAdapter;
  private taxiiAdapter?: TAXIIAdapter;
  private aggregator: IntelAggregator;
  private extractor: IOCExtractor;

  constructor() {
    const definition: HandDefinition = {
      id: 'threat-intel-collector',
      name: 'Threat Intel Collector Hand',
      description: 'Autonomous threat intelligence collection from MISP, OTX, TAXII',
      category: 'security',
      version: '1.0.0',
      requirements: [],
      settings: [
        {
          key: 'misp_url',
          label: 'MISP URL',
          description: 'MISP server URL',
          settingType: 'text',
          default: '',
        },
        {
          key: 'misp_api_key',
          label: 'MISP API Key',
          description: 'MISP API key for authentication',
          settingType: 'password',
          default: '',
        },
        {
          key: 'otx_api_key',
          label: 'OTX API Key',
          description: 'AlienVault OTX API key',
          settingType: 'password',
          default: '',
        },
        {
          key: 'update_frequency',
          label: 'Update Frequency',
          description: 'How often to collect intelligence',
          settingType: 'select',
          default: 'daily',
          options: [
            { value: 'hourly', label: 'Every hour' },
            { value: 'daily', label: 'Daily' },
            { value: 'weekly', label: 'Weekly' },
          ],
        },
      ],
      metrics: [
        { label: 'Intel Items Collected', memoryKey: 'intel_count', format: 'number' },
        { label: 'IOCs Extracted', memoryKey: 'ioc_count', format: 'number' },
        { label: 'Sources Active', memoryKey: 'sources_count', format: 'number' },
      ],
      tools: [
        'memory_store',
        'memory_recall',
        'knowledge_add_entity',
        'knowledge_add_relation',
        'knowledge_query',
        'schedule_create',
        'schedule_list',
        'schedule_delete',
      ],
      systemPrompt: `You are an autonomous threat intelligence collector. Your role is to:
1. Connect to multiple threat intelligence sources (MISP, OTX, TAXII)
2. Collect and aggregate threat intelligence data
3. Extract Indicators of Compromise (IOCs)
4. Update the knowledge graph with new intelligence
5. Report significant findings

Always verify the authenticity and relevance of collected intelligence.`,
    };

    super(definition);
    this.aggregator = new IntelAggregator();
    this.extractor = new IOCExtractor();
  }

  async initialize(): Promise<void> {
    const config = this.getConfig();

    // 初始化MISP适配器
    if (config.misp_url && config.misp_api_key) {
      this.mispAdapter = new MISPAdapter(config.misp_url, config.misp_api_key);
      await this.mispAdapter.connect();
    }

    // 初始化OTX适配器
    if (config.otx_api_key) {
      this.otxAdapter = new OTXAdapter(config.otx_api_key);
      await this.otxAdapter.connect();
    }

    // 初始化TAXII适配器
    if (config.taxii_url && config.taxii_api_key) {
      this.taxiiAdapter = new TAXIIAdapter(config.taxii_url, config.taxii_api_key);
      await this.taxiiAdapter.connect();
    }
  }

  async execute(context: HandContext): Promise<HandResult> {
    const startTime = Date.now();
    const intelData: IntelData[] = [];

    try {
      // 1. 从MISP收集情报
      if (this.mispAdapter) {
        this.reportProgress(10, 'Collecting from MISP...');
        const mispIntel = await this.mispAdapter.fetchEvents({
          limit: 100,
          since: context.lastRun,
        });
        intelData.push(...mispIntel);
      }

      // 2. 从OTX收集情报
      if (this.otxAdapter) {
        this.reportProgress(30, 'Collecting from OTX...');
        const otxIntel = await this.otxAdapter.getPulses({
          limit: 100,
          since: context.lastRun,
        });
        intelData.push(...otxIntel);
      }

      // 3. 从TAXII收集情报
      if (this.taxiiAdapter) {
        this.reportProgress(50, 'Collecting from TAXII...');
        const taxiiIntel = await this.taxiiAdapter.getObjects({
          limit: 100,
          since: context.lastRun,
        });
        intelData.push(...taxiiIntel);
      }

      // 4. 聚合情报
      this.reportProgress(70, 'Aggregating intelligence...');
      const aggregatedIntel = await this.aggregator.aggregate(intelData);

      // 5. 提取IOC
      this.reportProgress(80, 'Extracting IOCs...');
      const iocs = await this.extractor.extract(aggregatedIntel);

      // 6. 更新知识图谱
      this.reportProgress(90, 'Updating knowledge graph...');
      await this.updateKnowledgeGraph(iocs);

      // 7. 更新指标
      this.updateMetrics(intelData.length, iocs.length);

      return {
        success: true,
        data: {
          intelCount: intelData.length,
          iocCount: iocs.length,
          sources: this.getActiveSources(),
        },
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    } catch (error) {
      await this.handleError(error as Error);
      return {
        success: false,
        error: {
          code: 'INTEL_COLLECTION_FAILED',
          message: (error as Error).message,
          recoverable: true,
        },
        duration: Date.now() - startTime,
        metrics: this.getMetrics(),
      };
    }
  }

  async terminate(): Promise<void> {
    if (this.mispAdapter) {
      await this.mispAdapter.disconnect();
    }
    if (this.otxAdapter) {
      await this.otxAdapter.disconnect();
    }
    if (this.taxiiAdapter) {
      await this.taxiiAdapter.disconnect();
    }
  }

  private getActiveSources(): string[] {
    const sources: string[] = [];
    if (this.mispAdapter) sources.push('MISP');
    if (this.otxAdapter) sources.push('OTX');
    if (this.taxiiAdapter) sources.push('TAXII');
    return sources;
  }

  private updateMetrics(intelCount: number, iocCount: number): void {
    // 更新指标
    this.context.reportMetric('intel_count', intelCount);
    this.context.reportMetric('ioc_count', iocCount);
    this.context.reportMetric('sources_count', this.getActiveSources().length);
  }

  static getDefinition(): HandDefinition {
    return {
      id: 'threat-intel-collector',
      name: 'Threat Intel Collector Hand',
      description: 'Autonomous threat intelligence collection',
      category: 'security',
      version: '1.0.0',
      requirements: [],
      settings: [],
      metrics: [],
      tools: [],
      systemPrompt: '',
    };
  }
}
```

### ioc-extractor.ts 骨架

```typescript
import type { AggregatedIntel, IOC, IOCType } from './types';

export class IOCExtractor {
  // IP地址正则
  private readonly IP_REGEX = /\b(?:\d{1,3}\.){3}\d{1,3}\b/g;
  
  // 域名正则
  private readonly DOMAIN_REGEX = /\b(?:[a-z0-9](?:[a-z0-9-]{0,61}[a-z0-9])?\.)+[a-z]{2,}\b/gi;
  
  // MD5/SHA1/SHA256正则
  private readonly HASH_REGEX = {
    md5: /\b[a-fA-F0-9]{32}\b/g,
    sha1: /\b[a-fA-F0-9]{40}\b/g,
    sha256: /\b[a-fA-F0-9]{64}\b/g,
  };
  
  // URL正则
  private readonly URL_REGEX = /https?:\/\/[^\s<>"{}|\\^`\[\]]+/gi;
  
  // 邮箱正则
  private readonly EMAIL_REGEX = /\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b/g;

  async extract(intel: AggregatedIntel): Promise<IOC[]> {
    const iocs: IOC[] = [];

    for (const item of intel.items) {
      // 提取IP
      const ips = this.extractIPs(item.content);
      iocs.push(...ips.map(ip => this.createIOC('ip', ip, item)));

      // 提取域名
      const domains = this.extractDomains(item.content);
      iocs.push(...domains.map(domain => this.createIOC('domain', domain, item)));

      // 提取哈希
      const hashes = this.extractHashes(item.content);
      iocs.push(...hashes.map(hash => this.createIOC('hash', hash, item)));

      // 提取URL
      const urls = this.extractURLs(item.content);
      iocs.push(...urls.map(url => this.createIOC('url', url, item)));

      // 提取邮箱
      const emails = this.extractEmails(item.content);
      iocs.push(...emails.map(email => this.createIOC('email', email, item)));
    }

    return this.deduplicate(iocs);
  }

  extractIPs(content: string): string[] {
    const matches = content.match(this.IP_REGEX) || [];
    return this.filterPrivateIPs(matches);
  }

  extractDomains(content: string): string[] {
    const matches = content.match(this.DOMAIN_REGEX) || [];
    return matches;
  }

  extractHashes(content: string): Array<{ type: string; value: string }> {
    const hashes: Array<{ type: string; value: string }> = [];

    // MD5
    const md5 = content.match(this.HASH_REGEX.md5) || [];
    hashes.push(...md5.map(h => ({ type: 'md5', value: h })));

    // SHA1
    const sha1 = content.match(this.HASH_REGEX.sha1) || [];
    hashes.push(...sha1.map(h => ({ type: 'sha1', value: h })));

    // SHA256
    const sha256 = content.match(this.HASH_REGEX.sha256) || [];
    hashes.push(...sha256.map(h => ({ type: 'sha256', value: h })));

    return hashes;
  }

  extractURLs(content: string): string[] {
    return content.match(this.URL_REGEX) || [];
  }

  extractEmails(content: string): string[] {
    return content.match(this.EMAIL_REGEX) || [];
  }

  private createIOC(type: IOCType, value: string, source: any): IOC {
    return {
      id: `${type}-${value}`,
      type,
      value,
      source: source.id,
      timestamp: new Date(),
      confidence: source.confidence || 'medium',
      tags: source.tags || [],
    };
  }

  private filterPrivateIPs(ips: string[]): string[] {
    return ips.filter(ip => {
      const parts = ip.split('.').map(Number);
      // 过滤私有IP
      if (parts[0] === 10) return false;
      if (parts[0] === 172 && parts[1] >= 16 && parts[1] <= 31) return false;
      if (parts[0] === 192 && parts[1] === 168) return false;
      return true;
    });
  }

  private deduplicate(iocs: IOC[]): IOC[] {
    const seen = new Set<string>();
    return iocs.filter(ioc => {
      const key = `${ioc.type}-${ioc.value}`;
      if (seen.has(key)) return false;
      seen.add(key);
      return true;
    });
  }
}
```

```typescript
export interface IntelSource {

  enabled: boolean;

export interface IntelData {
  id: string;
  source: string;
  type: 'event' | 'pulse' | 'stix_object';

  content: string;
  timestamp: Date;
  tags: string[];
  confidence: 'low' | 'medium' | 'high';
  metadata: Record<string, unknown>;
}

export interface AggregatedIntel {
  id: string;
  items: IntelData[];
```
