## 示例代码片段

### scanner.ts 骨架

```typescript
import { PatternLibrary } from './patterns';
import { InjectionDetector } from './detector';
import { PromptSanitizer } from './sanitizer';
import { PromptAuditor } from './auditor';
import { ScanResult, InjectionAttempt, SecurityReport } from './types';

export class PromptScanner {
  private patternLibrary: PatternLibrary;
  private detector: InjectionDetector;
  private sanitizer: PromptSanitizer;
  private auditor: PromptAuditor;

  constructor() {
    this.patternLibrary = new PatternLibrary();
    this.detector = new InjectionDetector(this.patternLibrary);
    this.sanitizer = new PromptSanitizer();
    this.auditor = new PromptAuditor();
  }

  /**
   * 扫描提示
   */
  scan(prompt: string): ScanResult {
    // 检测注入尝试
    const attempts = this.detectInjection(prompt);
    
    // 计算风险评分
    const riskScore = this.calculateRiskScore(attempts);
    
    // 确定风险等级
    const riskLevel = this.determineRiskLevel(riskScore);
    
    // 记录审计
    this.auditor.audit(prompt, attempts, riskScore);
    
    return {
      prompt,
      attempts,
      riskScore,
      riskLevel,
      timestamp: new Date(),
    };
  }

  /**
   * 检测注入
   */
  detectInjection(prompt: string): InjectionAttempt[] {
    const attempts: InjectionAttempt[] = [];

    // 检测覆盖尝试
    const overrideAttempts = this.detector.detectOverride(prompt);
    attempts.push(...overrideAttempts);

    // 检测数据渗出
    const exfiltrationAttempts = this.detector.detectExfiltration(prompt);
    attempts.push(...exfiltrationAttempts);

    // 检测角色扮演
    const rolePlayAttempts = this.detector.detectRolePlay(prompt);
    attempts.push(...rolePlayAttempts);

    // 检测Shell引用
    const shellAttempts = this.detector.detectShellReference(prompt);
    attempts.push(...shellAttempts);

    return attempts;
  }

  /**
   * 计算风险评分
   */
  calculateRiskScore(attempts: InjectionAttempt[]): number {
    if (attempts.length === 0) return 0;

    // 根据严重性加权
    let score = 0;
    for (const attempt of attempts) {
      switch (attempt.severity) {
        case 'critical':
          score += 40;
          break;
        case 'high':
          score += 25;
          break;
        case 'medium':
          score += 15;
          break;
        case 'low':
          score += 5;
          break;
      }
    }

    return Math.min(100, score);
  }

  /**
   * 确定风险等级
   */
  private determineRiskLevel(score: number): 'critical' | 'high' | 'medium' | 'low' | 'safe' {
    if (score >= 80) return 'critical';
    if (score >= 60) return 'high';
    if (score >= 40) return 'medium';
    if (score >= 20) return 'low';
    return 'safe';
  }

  /**
   * 生成报告
   */
  generateReport(result: ScanResult): SecurityReport {
    return {
      prompt: result.prompt,
      riskLevel: result.riskLevel,
      riskScore: result.riskScore,
      attempts: result.attempts,
      recommendations: this.generateRecommendations(result),
      timestamp: result.timestamp,
    };
  }

  /**
   * 生成建议
   */
  private generateRecommendations(result: ScanResult): string[] {
    const recommendations: string[] = [];

    if (result.attempts.length > 0) {
      recommendations.push('Consider sanitizing the prompt before processing');
    }

    if (result.riskLevel === 'critical' || result.riskLevel === 'high') {
      recommendations.push('Reject or quarantine this prompt');
    }

    if (result.attempts.some(a => a.type === 'override')) {
      recommendations.push('Review system prompt boundaries');
    }

    if (result.attempts.some(a => a.type === 'exfiltration')) {
      recommendations.push('Check for data leakage risks');
    }

    return recommendations;
  }
}
```

```typescript
import { Pattern, PatternCategory } from './types';

export class PatternLibrary {
  private patterns: Map<string, Pattern>;

  constructor() {
    this.patterns = new Map();
    this.loadDefaultPatterns();
  }

  /**
   * 加载默认模式
   */
  private loadDefaultPatterns(): void {
    // 覆盖尝试模式
    this.addPattern({
      id: 'override-1',
      category: 'override',
      pattern: /ignore (all )?(previous|above) instructions/i,
      severity: 'critical',
      description: 'Attempt to override previous instructions',
    });

    this.addPattern({
      id: 'override-2',
      category: 'override',
      pattern: /disregard (all )?(previous|above)/i,
      severity: 'critical',
      description: 'Attempt to disregard instructions',
    });

    this.addPattern({
      id: 'override-3',
      category: 'override',
      pattern: /forget (all )?(previous|above)/i,
      severity: 'critical',
      description: 'Attempt to forget instructions',
    });

    // 角色扮演模式
    this.addPattern({
      id: 'roleplay-1',
      category: 'roleplay',
      pattern: /you are (now|a|an) /i,
      severity: 'high',
      description: 'Role play attempt',
    });

    this.addPattern({
      id: 'roleplay-2',
      category: 'roleplay',
      pattern: /act as (if|a|an) /i,
      severity: 'high',
      description: 'Role play attempt',
    });

    this.addPattern({
      id: 'roleplay-3',
      category: 'roleplay',
      pattern: /pretend (to be|that)/i,
      severity: 'high',
      description: 'Role play attempt',
    });

    // 数据渗出模式
    this.addPattern({
      id: 'exfiltration-1',
      category: 'exfiltration',
      pattern: /output (all |your )?(instructions|prompt|system)/i,
      severity: 'critical',
      description: 'Attempt to extract system prompt',
    });

    this.addPattern({
      id: 'exfiltration-2',
      category: 'exfiltration',
      pattern: /print (your |the )?(instructions|prompt)/i,
      severity: 'critical',
      description: 'Attempt to print instructions',
    });

    this.addPattern({
      id: 'exfiltration-3',
      category: 'exfiltration',
      pattern: /repeat (your |the )?(instructions|prompt)/i,
      severity: 'critical',
      description: 'Attempt to repeat instructions',
    });

    // Shell引用模式
    this.addPattern({
      id: 'shell-1',
      category: 'shell',
      pattern: /\$\{[^}]+\}/,
      severity: 'high',
      description: 'Shell variable reference',
    });

    this.addPattern({
      id: 'shell-2',
      category: 'shell',
      pattern: /`[^`]+`/,
      severity: 'medium',
      description: 'Shell command substitution',
    });

    this.addPattern({
      id: 'shell-3',
      category: 'shell',
      pattern: /\$\([^)]+\)/,
      severity: 'high',
      description: 'Shell command substitution',
    });
  }

  /**
   * 添加模式
   */
  addPattern(pattern: Pattern): void {
    this.patterns.set(pattern.id, pattern);
  }

  /**
   * 匹配模式
   */
  matchPatterns(text: string): Pattern[] {
    const matched: Pattern[] = [];

    for (const pattern of this.patterns.values()) {
      if (pattern.pattern.test(text)) {
        matched.push(pattern);
      }
    }

    return matched;
  }

  /**
   * 按类别获取模式
   */
  getPatternsByCategory(category: PatternCategory): Pattern[] {
    return Array.from(this.patterns.values())
      .filter(p => p.category === category);
  }
}
```

### sanitizer.ts 骨架

```typescript
import { ScanResult } from './types';

export class PromptSanitizer {
  /**
   * 清理提示
   */
  sanitize(prompt: string, scanResult: ScanResult): string {
    let sanitized = prompt;

    // 移除注入内容
    for (const attempt of scanResult.attempts) {
      sanitized = this.removeInjection(sanitized, attempt.match);
    }

    // 转义特殊字符
    sanitized = this.escapeSpecialChars(sanitized);

    // 中和威胁
    sanitized = this.neutralize(sanitized);

    return sanitized;
  }

  /**
   * 转义特殊字符
   */
  escapeSpecialChars(text: string): string {
    // 转义Shell特殊字符
    return text
      .replace(/\$/g, '\\$')
      .replace(/`/g, '\\`')
      .replace(/\(/g, '\\(')
      .replace(/\)/g, '\\)')
      .replace(/\{/g, '\\{')
      .replace(/\}/g, '\\}');
  }

  /**
   * 移除注入
   */
  removeInjection(text: string, match: string): string {
    // 移除匹配的注入内容
    return text.replace(match, '[REDACTED]');
  }

  /**
   * 中和威胁
   */
  neutralize(text: string): string {
    // 中和潜在威胁
    return text
      .replace(/ignore/gi, '[IGNORE]')
      .replace(/disregard/gi, '[DISREGARD]')
      .replace(/forget/gi, '[FORGET]');
  }
}
```

```typescript
// 模式类别
export type PatternCategory = 'override' | 'roleplay' | 'exfiltration' | 'shell';

// 严重性
export type Severity = 'critical' | 'high' | 'medium' | 'low';

// 模式
export interface Pattern {
  id: string;
  category: PatternCategory;
  pattern: RegExp;
  severity: Severity;
  description: string;
}

// 注入尝试
export interface InjectionAttempt {
  type: PatternCategory;
  pattern: Pattern;
  match: string;
  severity: Severity;
  position: {
    start: number;
    end: number;
  };
}

// 扫描结果
export interface ScanResult {
  prompt: string;
  attempts: InjectionAttempt[];
  riskScore: number;
  riskLevel: 'critical' | 'high' | 'medium' | 'low' | 'safe';
  timestamp: Date;
}

// 安全报告
export interface SecurityReport {
  prompt: string;
  riskLevel: string;
  riskScore: number;
  attempts: InjectionAttempt[];
  recommendations: string[];
  timestamp: Date;
}
```
