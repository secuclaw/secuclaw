## 示例代码片段

### doctor.ts 骨架

```typescript
import { checkConfig } from "./checks/config.js";
import { checkConnectivity } from "./checks/connectivity.js";
import { checkSecurity } from "./checks/security.js";
import { checkDependencies } from "./checks/dependencies.js";
import { checkApiKeys } from "./checks/api-keys.js";
import { checkMemory } from "./checks/memory.js";

export type DoctorResult = {
  check: string;
  status: "ok" | "warning" | "error";
  message: string;
  details?: unknown;
  fixable: boolean;
  fix?: () => Promise<void>;
};

export type DoctorCommandOpts = {
  fix?: boolean;
  json?: boolean;
  quiet?: boolean;
  checks?: string[];
};

export class DoctorCommand {
  private checks: Map<string, () => Promise<DoctorResult>>;
  
  constructor(private opts: DoctorCommandOpts = {}) {
    this.checks = new Map([
      ["config", checkConfig],
      ["connectivity", checkConnectivity],
      ["security", checkSecurity],
      ["dependencies", checkDependencies],
      ["api-keys", checkApiKeys],
      ["memory", checkMemory],
    ]);
  }
  
  async run(): Promise<DoctorResult[]> {
    const results: DoctorResult[] = [];
    const toRun = this.opts.checks ?? Array.from(this.checks.keys());
    
    for (const name of toRun) {
      const check = this.checks.get(name);
      if (!check) continue;
      
      if (!this.opts.quiet) {
        console.log(`Checking ${name}...`);
      }
      
      const result = await check();
      results.push(result);
      
      if (this.opts.fix && result.fixable && result.fix) {
        await result.fix();
      }
    }
    
    return results;
  }
  
  async fixAuto(): Promise<void> {
    const results = await this.run();
    for (const result of results) {
      if (result.fixable && result.fix) {
        await result.fix();
      }
    }
  }
}
```

### checks/config.ts

```typescript
import fs from "node:fs/promises";
import path from "node:path";
import { loadConfig } from "@secuclaw/core/config";

export async function checkConfig(): Promise<DoctorResult> {
  const configPath = path.join(process.env.HOME!, ".secuclaw", "config.json");
  
  // 检查配置文件是否存在
  try {
    await fs.stat(configPath);
  } catch {
    return {
      check: "config",
      status: "error",
      message: "Config file not found",
      details: { path: configPath },
      fixable: true,
      fix: async () => {
        await fs.mkdir(path.dirname(configPath), { recursive: true });
        await fs.writeFile(configPath, JSON.stringify({}, null, 2));
      },
    };
  }
  
  // 检查配置格式
  try {
    const config = await loadConfig();
    
    // 检查必需字段
    if (!config.agent?.model) {
      return {
        check: "config",
        status: "warning",
        message: "No default model configured",
        fixable: true,
      };
    }
    
    return {
      check: "config",
      status: "ok",
      message: "Configuration is valid",
    };
  } catch (err) {
    return {
      check: "config",
      status: "error",
      message: `Invalid config: ${err}`,
      fixable: false,
    };
  }
}
```
